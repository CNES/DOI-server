93      @Test
94      public void testApiWithHttp() throws IOException {
95          String port = DoiSettings.getInstance().getString(Consts.SERVER_HTTP_PORT);        
96          ClientResource client = new ClientResource("http://localhost:"+port+"/");
97          client.setChallengeResponse(ChallengeScheme.HTTP_BASIC, "admin", "admin");
98          Representation repApi = client.options();
99          String txt = repApi.getText();
100         client.release();
101         assertTrue("WADL API through HTTP", txt!=null && !txt.isEmpty());
102     }
