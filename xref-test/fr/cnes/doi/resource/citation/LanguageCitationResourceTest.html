103         String expResult = "af-ZA";

106         ClientResource client = new ClientResource("https://localhost:" + port + "/citation/language");
107         client.setNext(cl);
108         try {
109             List<String> rep = client.get(List.class);
110             result = rep.get(0);
111         } catch (ResourceException ex) {
112         } finally {
113             client.release();
