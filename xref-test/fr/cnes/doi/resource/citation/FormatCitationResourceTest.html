115     /**
116      * Test of getFormat method, of class FormatCitationResource.
117      */
118     @Test
119     public void testGetFormatHttps() {
120 	spec.createSpec(CrossCiteSpec.Spec.GET_FORMAT_200);
121 
122 	String expResult = CrossCiteSpec.Spec.GET_FORMAT_200.getBody();
123 	String result = "";
124 	String doiName = "10.1145/2783446.2783605";
125 	String style = "academy-of-management-review";
126 	String language = "af-ZA";
127 	String port = DoiSettings.getInstance().getString(Consts.SERVER_HTTPS_PORT);
128 	ClientResource client = new ClientResource(
129 		"https://localhost:" + port + "/citation/format");
130 	client.addQueryParameter("doi", doiName);
131 	client.addQueryParameter("lang", language);
132 	client.addQueryParameter("style", style);
133 	client.setNext(cl);
134 	Representation rep = client.get();
135 	try {
136 	    result = rep.getText();
137 	} catch (IOException ex) {
138 	}
139 	client.release();
140 	assertEquals("Test the citation format through a HTTPS server", expResult, result);
141 
142 	spec.verifySpec(CrossCiteSpec.Spec.GET_FORMAT_200);
143 
144     }
