122     /**
123      * Test of getMetadata method throw a HTTPS server, of class
124      * MetadataResource. A Status.SUCCESS_OK is expected when the metadata is
125      * available whereas a Status.CLIENT_ERROR_GONE is expected when the
126      * metadata is deleted.
127      *
128      * @throws java.io.IOException - if OutOfMemoryErrors
129      * @throws javax.xml.bind.JAXBException - if a parsing problem occurs
130      * @throws org.xml.sax.SAXException - if a parsing problme occurs
131      */
132     @Test
133     public void testGetMetadata() throws IOException, JAXBException, SAXException {
134         mdsServerStub.createSpec(MdsSpec.Spec.GET_METADATA_200);
135 
136         String port = DoiSettings.getInstance().getString(Consts.SERVER_HTTPS_PORT);
137         ClientResource client = new ClientResource("https://localhost:" + port + METADATA_SERVICE + MdsSpec.Spec.GET_METADATA_200.getTemplatePath());
138         client.setNext(cl);
139         int code;
140         String doi;
141         try {            
142             Resource resource = client.get(Resource.class);
143             code = client.getStatus().getCode();
144             doi = resource.getIdentifier().getValue();
145         } catch (ResourceException ex) {
146             code = ex.getStatus().getCode();
147             doi = "";
148         }
149         client.release();
150         assertTrue(MdsSpec.Spec.GET_METADATA_200.getStatus() == code);
151         assertTrue(MdsSpec.Spec.GET_METADATA_200.getTemplatePath().equals(doi) || doi.isEmpty());
152 
153         mdsServerStub.verifySpec(MdsSpec.Spec.GET_METADATA_200);
154     }

264         ClientResource client = new ClientResource("https://localhost:" + port + METADATA_SERVICE +MdsSpec.Spec.DELETE_METADATA_200.getTemplatePath());

266         client.setChallengeResponse(new ChallengeResponse(ChallengeScheme.HTTP_BASIC, "malapert", "pwd"));

268         Map<String, Object> reqAttribs = client.getRequestAttributes();
269         Series headers = (Series) reqAttribs.get(RESTLET_HTTP_HEADERS);
270         if (headers == null) {
271             headers = new Series<>(Header.class);
272             reqAttribs.put(RESTLET_HTTP_HEADERS, headers);
273         }
274         headers.add(UtilsHeader.SELECTED_ROLE_PARAMETER, "828606");
275         int code;
276         try {
277             Representation rep = client.delete();
278             code = client.getStatus().getCode();
279             rep.exhaust();
280         } catch (ResourceException ex) {
281             code = ex.getStatus().getCode();
282         }
283         client.release();
