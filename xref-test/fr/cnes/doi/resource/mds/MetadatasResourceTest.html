293     private void testSpecCreateMetadataAsObj(MdsSpec.Spec spec, InputStream is, String login, String pwd, String role, int exactly) throws IOException {        
294         // Creates the MetadataStoreService stub        
295         mdsServerStub.createSpec(spec);
296 
297         result = new BufferedReader(new InputStreamReader(is)).lines().collect(Collectors.joining("\n"));
298         String port = DoiSettings.getInstance().getString(Consts.SERVER_HTTPS_PORT);
299         ClientResource client = new ClientResource("https://localhost:" + port + METADATA_SERVICE);
300         client.setNext(cl);
301         client.setChallengeResponse(new ChallengeResponse(ChallengeScheme.HTTP_BASIC, login, pwd));
302         final String RESTLET_HTTP_HEADERS = "org.restlet.http.headers";
303         Map<String, Object> reqAttribs = client.getRequestAttributes();
304         Series headers = (Series) reqAttribs.get(RESTLET_HTTP_HEADERS);
305         if (headers == null) {
306             headers = new Series<>(Header.class);
307             reqAttribs.put(RESTLET_HTTP_HEADERS, headers);
308         }
309         if (role != null) {
310             headers.add(UtilsHeader.SELECTED_ROLE_PARAMETER, role);
311         }
312         int code;
313         try {
314             Representation rep = client.post(new StringRepresentation(result, MediaType.APPLICATION_XML));
315             code = client.getStatus().getCode();
316             rep.exhaust();
317         } catch (ResourceException ex) {
318             code = ex.getStatus().getCode();
319         } finally {
320             client.release();    
321         }        
322         assertEquals(spec.getStatus(), code);        
323         
324         // Checks the stub.        
325         if (exactly == -1) {
326             mdsServerStub.verifySpec(spec);       
327         } else {
328             mdsServerStub.verifySpec(spec, exactly);   
329         }
