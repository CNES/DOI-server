104         
105         String port = DoiSettings.getInstance().getString(Consts.SERVER_HTTP_PORT);
106         ClientResource client = new ClientResource("http://localhost:"+port+DOIS_SERVICE+MdsSpec.Spec.GET_DOI_200.getTemplatePath());

108         Status status = client.getStatus();
109         assertEquals("Test the status code is the right one", MdsSpec.Spec.GET_DOI_200.getStatus(), status.getCode());
110         assertEquals("Test the landing page is the right one", MdsSpec.Spec.GET_DOI_200.getBody(), rep.getText());
111         
112         mdsServerStub.verifySpec(MdsSpec.Spec.GET_DOI_200);

137     /**
138      * Test of getDoi method when the DOI is not found through a HTTPS server, of class DoiResource.
139      * A CLIENT_ERROR_NOT_FOUND status is expected because the DOI does not exist.
140      * @throws java.io.IOException - if OutOfMemoryErrors
141      */
142     @Test
143     public void testGetDoiNotFoundHttps() throws IOException {            
144         mdsServerStub.createSpec(MdsSpec.Spec.GET_DOI_404);
145 
146         String port = DoiSettings.getInstance().getString(Consts.SERVER_HTTPS_PORT);
147         ClientResource client = new ClientResource("https://localhost:"+port+DOIS_SERVICE+MdsSpec.Spec.GET_DOI_404.getTemplatePath());
148         client.setNext(cl);
149         int code;
150         try {
151             Representation rep = client.get();
152             code = client.getStatus().getCode();
153             rep.exhaust();
154         } catch (ResourceException ex) {
155             code = ex.getStatus().getCode();
156         } 
157         client.release();
158         assertEquals(MdsSpec.Spec.GET_DOI_404.getStatus(), code);
159         
160         mdsServerStub.verifySpec(MdsSpec.Spec.GET_DOI_404);
161        
162     }
