177     @Test
178     public void testCreateDOIs() {
179         long startTime = System.currentTimeMillis();
180         ConcurrentHashMap map = new ConcurrentHashMap();
181         map.put("nbErrors", 0);
182         mdsServerStub.createSpec(MdsSpec.Spec.POST_METADATA_201);
183         mdsServerStub.createSpec(MdsSpec.Spec.POST_DOI_201);
184

186         testMultiThreads(CreateDOI.class, map, NB_ITERS);        
187         clientExec.shutdown();
188 
189         try {
190             clientExec.awaitTermination(Long.MAX_VALUE, TimeUnit.MILLISECONDS);
191         } catch (InterruptedException e) {
192             LOG.log(Level.SEVERE, null, e);
193         }
194         
195         long stopTime = System.currentTimeMillis();
196         long elapsedTime = stopTime - startTime;
197         double meanProcessingTime = elapsedTime / (NB_ITERS - (int) map.get("nbErrors"));
198 
199         double expectedTime = 5.0 / 100.0 * 1000.0; //1 s per DOI
200         LOG.log(Level.INFO, "All working fine : Mean request processing time {0} ms with {1} error, expected time {2} ms", new Object[]{meanProcessingTime, (int) map.get("nbErrors"), expectedTime});        
201         Assert.assertTrue("Test the performances of DOIs creation",  meanProcessingTime <= expectedTime);
202     }

232     private void testMultiThreads(final Class jobTask, final ConcurrentHashMap map, int nbIters) {
233         try {
234             for (int i = 0; i < nbIters; i++) {
235                 try {
236                     JobTask task = (JobTask) jobTask.newInstance();
237                     task.setParameters(map, this.metadata);
238                     clientExec.execute((Runnable) task);
239                 } catch (InstantiationException | IllegalAccessException ex) {
240                     LOG.log(Level.SEVERE, null, ex);
241                     int nbErrors = (int) map.get("nbErrors");
242                     map.put("nbErrors", nbErrors + 1);
243                     break;
244                 }
245             }
246         } catch (RuntimeException ex) {
247             int nbErrors = (int) map.get("nbErrors");
248             map.put("nbErrors", nbErrors + 1); 
249             LOG.log(Level.SEVERE, null, ex);
250         }
251 
252     }

254     private static class CreateDOI implements Runnable, JobTask {
255 
256         private ConcurrentHashMap map;
257         private String doiMetadata;
258 
259         public CreateDOI() {
260         }

262         @Override
263         public void run() {
264             
265             final String userAdmin = "malapert";
266             final String password = "pwd";
267             
268             Form doiForm = new Form();
269             doiForm.add(new Parameter(DoisResource.DOI_PARAMETER, "10.5072/828606/8c3e91ad45ca855b477126bc073ae44b"));
270             doiForm.add(new Parameter(DoisResource.URL_PARAMETER, "http://www.cnes.fr"));
271 
272             String port = DoiSettings.getInstance().getString(Consts.SERVER_HTTP_PORT);
273             ClientResource client = new ClientResource("http://localhost:" + port + DOIS_SERVICE);
274             client.setChallengeResponse(new ChallengeResponse(ChallengeScheme.HTTP_BASIC, userAdmin, password));
275             final String RESTLET_HTTP_HEADERS = "org.restlet.http.headers";
276 
277             Map<String, Object> reqAttribs = client.getRequestAttributes();
278             Series headers = (Series) reqAttribs.get(RESTLET_HTTP_HEADERS);
279             if (headers == null) {
280                 headers = new Series<>(Header.class);
281                 reqAttribs.put(RESTLET_HTTP_HEADERS, headers);
282             }
283             headers.add(UtilsHeader.SELECTED_ROLE_PARAMETER, "828606");
284             int code;
285             try {
286                 Representation rep = client.post(doiForm);
287                 rep.exhaust();
288                 code = client.getStatus().getCode();
289             } catch (ResourceException ex) {
290                 code = ex.getStatus().getCode();
291                 int nbErrors = (int) this.map.get("nbErrors");
292                 this.map.put("nbErrors", nbErrors + 1);
293             } catch (IOException ex) {
294                 int nbErrors = (int) this.map.get("nbErrors");
295                 this.map.put("nbErrors", nbErrors + 1);
296             } finally {
297                 client.release();
298             }               
299 
300             client = new ClientResource("http://localhost:" + port + METADATA_SERVICE);
301             client.setNext(cl);
302             client.setChallengeResponse(new ChallengeResponse(ChallengeScheme.HTTP_BASIC, userAdmin, password));
303             reqAttribs = client.getRequestAttributes();
304             headers = (Series) reqAttribs.get(RESTLET_HTTP_HEADERS);
305             if (headers == null) {
306                 headers = new Series<>(Header.class);
307                 reqAttribs.put(RESTLET_HTTP_HEADERS, headers);
308             }
309             headers.add(UtilsHeader.SELECTED_ROLE_PARAMETER, "828606");
310 
311             try {
312                 Representation rep = client.post(new StringRepresentation(this.doiMetadata, MediaType.APPLICATION_XML));
313                 code = client.getStatus().getCode();
314                 rep.exhaust();
315             } catch (ResourceException ex) {
316                 code = ex.getStatus().getCode();
317                 int nbErrors = (int) this.map.get("nbErrors");
318                 this.map.put("nbErrors", nbErrors + 1);
319             } catch (IOException ex) {
320                 int nbErrors = (int) this.map.get("nbErrors");
321                 this.map.put("nbErrors", nbErrors + 1);            
322             } finally {
323                 client.release();
324             }
325         }
