160     /**
161      * Stops the server.
162      */
163     public static void close() {
164         try {
165             for (DOIUser user : das.getAllDOIusers()) {
166                 das.removeDOIUser(user.getUsername());
167             }
168             for (DOIProject project : das.getAllDOIProjects()) {
169                 das.removeDOIProject(project.getSuffix());
170             }
171             for (String token : das.getTokens()) {
172                 das.deleteToken(token);
173             }
174             
175         } catch (DOIDbException e) {
176             fail("Please create the database with the following informations:\nURL:" + DoiSettings.
177                     getInstance().getString(DB_URL) + "\nUser: " + DoiSettings.getInstance().
178                     getString(DB_USER) + "\nPwd: " + DoiSettings.getInstance().getSecret(DB_PWD));
179         }
180     }
