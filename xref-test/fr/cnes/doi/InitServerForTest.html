175     public static void testTitle(final String title, final ColorStatus status,
176 	    final String skipMessage) {
177 	final String colorStatus;
178 	switch (status) {
179 	    case SKIP:
180 		colorStatus = ANSI_YELLOW + ColorStatus.SKIP.name().toLowerCase() + ANSI_RESET;
181 		break;
182 	    case FAILED:
183 		colorStatus = ANSI_RED + ColorStatus.FAILED.name().toLowerCase() + ANSI_RESET;
184 		break;
185 	    default:
186 		colorStatus = ANSI_GREEN + ColorStatus.OK.name().toLowerCase() + ANSI_RESET;
187 	}
188 
189 	String pt = String.join("", Collections.nCopies(80, "."));
190 	if (title.length() > pt.length()) {
191 	    pt = "";
192 	} else {
193 	    pt = pt.substring(title.length() - 1, pt.length() - 1);
194 	}
195 
196 	String skipMessageDisplay = (skipMessage == null) ? ""
197 		: ANSI_PURPLE + "  (" + skipMessage + ")" + ANSI_RESET;
198 	System.out.println(ANSI_BLUE + "Testing: " + title + ANSI_RESET + pt + colorStatus
199 		+ skipMessageDisplay);
200     }

238 	    System.out
