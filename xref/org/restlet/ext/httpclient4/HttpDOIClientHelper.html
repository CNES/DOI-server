191     /**

194     @Override
195     public ClientCall create(final Request request) {
196         ClientCall result = null;
197 
198         try {
199             result = new HttpMethodCall(this, request.getMethod().toString(),
200                     ReferenceUtils.update(request.getResourceRef(), request)
201                             .toString(), request.isEntityAvailable());
202         } catch (IOException ioe) {
203             getLogger().log(Level.WARNING,
204                     "Unable to create the HTTP client call", ioe);
205         }
206 
207         return result;
208     }

219     /**
220      * Configures the http client.
221      *
222      * @param parameters parameters
223      */
224     private void configure(final Series<Parameter> parameters) {
225         final Map<String, String> config = new HashMap<>();
226         config.put(HttpClient.CONNECTION_MAX_PER_ROUTE, String.valueOf(this.
227                 getMaxConnectionsPerHost()));
228         config.put(HttpClient.CONNECTION_MAX_TOTAL, String.valueOf(this.getMaxTotalConnections()));
229         config.put(HttpClient.CONNECTION_TIME_TO_LIVE_MS, String.valueOf(this.getIdleTimeout()));
230         config.put(HttpClient.MAX_RETRY, String.valueOf(this.getRetry()));
231         config.put(HttpClient.RETRY_DELAY, String.valueOf(this.getRetryDelay()));
232         config.put(HttpClient.MAX_REDIRECTION, String.valueOf(this.getMaxRedirects()));
233         config.computeIfAbsent(HttpClient.KEYSTORE_TYPE, v -> this.getKeyStoreType());
234         config.computeIfAbsent(HttpClient.KEYSTORE_PATH, v -> this.getKeyStorePath());
235         config.computeIfAbsent(HttpClient.KEYSTORE_PWD, v -> this.getKeyStorePwd());
236         config.computeIfAbsent(HttpClient.TRUSTSTORE_TYPE, v -> this.getTrustStoreType());
237         config.computeIfAbsent(HttpClient.TRUSTSTORE_PATH, v -> this.getTrustStorePath());
238         config.computeIfAbsent(HttpClient.TRUSTSTORE_PWD, v -> this.getTrustStorePwd());
239 
240         LOG.info("Http Config : {}", config);
241 
242         final Type type = HttpClientFactory.Type.valueOf(ProxySettings.getInstance().getProxyType());
243         ProxySettings.getInstance().configureProxy();
244         LOG.info("Httpclient type : {}", type);
245         this.httpClient = HttpClientFactory.create(type, this.isDisabledSSL(), config);
246     }

264             this.getHttpClient().close();
265             this.httpClient = null;
266             LOG.info("Stopping the internal HTTP client");
267         }
