102     /**
103      * Loads configuration file and set it in memory.
104      *
105      * @param properties Configuration file
106      * @param level LOG level
107      */
108     private void init(final Properties properties, Level level) {
109         if (Level.OFF.equals(level)) {
110             level = null; // this fixes a bug in log4j2
111         }
112         LOG.traceEntry("Parameter\n\tproperties:{}\n\tlevel:{}", properties, level);
113         LOG.log(level, "----- DOI parameters ----");
114         fillConcurrentMap(properties, level);
115         computePathOfTheApplication();
116         PluginFactory.init(DoiSettings.MAP_PROPERTIES);
117         final Enumeration<String> keys = DoiSettings.MAP_PROPERTIES.keys();
118         while (keys.hasMoreElements()) {
119             final String key = keys.nextElement();
120             LOG.log(
121                     level, "{} = {}",
122                     key,
123                     isPassword(String.valueOf(key))
124                     ? Utils.transformPasswordToStars(String.valueOf(DoiSettings.MAP_PROPERTIES.get(
125                             key)))
126                     : DoiSettings.MAP_PROPERTIES.get(key)
127             );
128         }
129         LOG.log(level, "DOI settings have been loaded");
130         LOG.log(level, "-------------------------");
131         LOG.log(level, properties.getProperty(Consts.NAME) + " loaded");
132         LOG.traceExit();
133     }

135     /**
136      * Computes path of the application.
137      */
138     private void computePathOfTheApplication() {
139         LOG.traceEntry();
140         try {
141             final String path = Starter.class.getProtectionDomain().getCodeSource().getLocation().
142                     getPath();
143             String decodedPath = URLDecoder.decode(path, "UTF-8");
144             final int posLastSlash = decodedPath.lastIndexOf('/');
145             decodedPath = decodedPath.substring(0, posLastSlash);
146             this.pathApp = decodedPath;
147         } catch (UnsupportedEncodingException ex) {
148             throw LOG.throwing(new DoiRuntimeException(ex));
149         }
150         LOG.traceExit();
151     }

153     /**
154      * Load configuration file.
155      *
156      * @return the configuration file content
157      */
158     private Properties loadConfigurationFile() {
159         LOG.traceEntry();
160         final Properties properties = new DOIProperties();
161         final ClientResource client = new ClientResource(LocalReference.createClapReference(
162                 "class/config.properties"));
163         final Representation configurationFile = client.get();
164         try {
165             LOG.debug("Loading " + CONFIG_PROPERTIES + " by default");
166             properties.load(configurationFile.getStream());
167         } catch (IOException e) {
168             LOG.fatal("Unable to load " + CONFIG_PROPERTIES);
169             throw LOG.throwing(new DoiRuntimeException("Unable to load class/config.properties", e));
170         } finally {
171             client.release();
172         }
173         return LOG.traceExit(properties);
174     }

178      */
179     public void validConfigurationFile() {
180         LOG.traceEntry();
181         final StringBuilder validation = new StringBuilder();
182         final String message = "Sets ";
183         if (isNotExist(DoiSettings.MAP_PROPERTIES, Consts.INIST_DOI)) {
184             validation.append(message).append(Consts.INIST_DOI).append("\n");
185         }
186         if (isNotExist(DoiSettings.MAP_PROPERTIES, Consts.INIST_LOGIN)) {
187             validation.append(message).append(Consts.INIST_LOGIN).append("\n");
188         }
189         if (isNotExist(DoiSettings.MAP_PROPERTIES, Consts.INIST_PWD)) {
190             validation.append(message).append(Consts.INIST_PWD).append("\n");
191         }
192         if (isNotExist(DoiSettings.MAP_PROPERTIES, Consts.SERVER_PROXY_TYPE)) {
193             validation.append(message).append(Consts.SERVER_PROXY_TYPE).append("\n");
194         }
195         if (isNotExist(DoiSettings.MAP_PROPERTIES, Consts.PLUGIN_PROJECT_SUFFIX)) {
196             validation.append(message).append(Consts.PLUGIN_PROJECT_SUFFIX).append("\n");
197         }
198         if (isNotExist(DoiSettings.MAP_PROPERTIES, Consts.PLUGIN_TOKEN)) {
199             validation.append(message).append(Consts.PLUGIN_TOKEN).append("\n");
200         }
201         if (isNotExist(DoiSettings.MAP_PROPERTIES, Consts.PLUGIN_USER_GROUP_MGT)) {
202             validation.append(message).append(Consts.PLUGIN_USER_GROUP_MGT).append("\n");
203         }
204 
205         if (isNotExist(DoiSettings.MAP_PROPERTIES, Consts.PLUGIN_AUTHENTICATION)) {
206             validation.append(message).append(Consts.PLUGIN_AUTHENTICATION).append("\n");
207         }
208 
209         validation.append(PluginFactory.getAuthenticationSystem().validate());
210         validation.append(PluginFactory.getProjectSuffix().validate());
211         validation.append(PluginFactory.getToken().validate());
212         validation.append(PluginFactory.getUserManagement().validate());
213 
214         if (validation.length() != 0) {
215             throw LOG.traceExit(new DoiRuntimeException(validation.toString()));
216         }
217         LOG.traceExit();
218     }

252     private void fillConcurrentMap(final Properties properties, final Level level) {
253         LOG.traceEntry("Parameters\n\tproperties : {}\n\tlevel : {}", properties, level);

261      * Tests if the value of the key is a password
262      *
263      * @param key key to test
264      * @return True when the value of the key is a password otherwise false
265      */
266     private boolean isPassword(final String key) {
267         LOG.traceEntry("Parameter\n\tkey:{}", key);
268         return LOG.traceExit(Consts.INIST_PWD.equals(key)
269                 || PluginFactory.isPassword(Consts.PLUGIN_AUTHENTICATION, key)
270                 || PluginFactory.isPassword(Consts.PLUGIN_PROJECT_SUFFIX, key)
271                 || PluginFactory.isPassword(Consts.PLUGIN_TOKEN, key)
272                 || PluginFactory.isPassword(Consts.PLUGIN_USER_GROUP_MGT, key)
273                 || Consts.SERVER_HTTPS_KEYSTORE_PASSWD.equals(key)
274                 || Consts.SERVER_HTTPS_SECRET_KEY.equals(key) || Consts.SERVER_PROXY_PWD.equals(key)
275                 || Consts.SMTP_AUTH_PWD.equals(key) || Consts.TOKEN_KEY.equals(key));
276     }

322     /**
323      * Returns the value of the key or null if no mapping for the key. A special processing
324      * inputStream done for the key {@value fr.cnes.doi.settings.Consts#INIST_DOI}. When this key
325      * inputStream called the value changes in respect of
326      * {@value fr.cnes.doi.settings.Consts#CONTEXT_MODE}. When
327      * {@value fr.cnes.doi.settings.Consts#CONTEXT_MODE} inputStream set to PRE_PROD,
328      * {@value fr.cnes.doi.settings.Consts#INIST_DOI} inputStream set to {@value #INIST_TEST_DOI}.
329      *
330      * @param key key to search
331      * @return the value of the key orl null if does not exist
332      */
333     public String getString(final String key) {
334         LOG.traceEntry("Parameter\n\tkey : {}", key);
335         final String value;
336         if (Consts.INIST_DOI.equals(key)) {
337             final String context = this.getString(Consts.CONTEXT_MODE, "DEV");
338             value = "PRE_PROD".equals(context) ? INIST_TEST_DOI : this.getString(Consts.INIST_DOI,
339                     null);
340         } else {
341             value = this.getString(key, null);
342         }
343         return LOG.traceExit(value);
344     }

346     /**
347      * Returns the decoded value of the key. An exception inputStream raised when the key
348      * inputStream not encoded on 16bits.
349      *
350      * @param key key to search
351      * @return the decoded vale
352      */
353     public String getSecret(final String key) {
354         LOG.traceEntry("Parameter\n\tkey : {}", key);
355         final String result;
356         final String value = getString(key, "");
357         final boolean isEncrypted = Boolean.parseBoolean(
358                 this.getString(Consts.ENCRYPTED_FIELDS, "true"));
359         if (Utils.isEmpty(value)) {
360             result = value;
361         } else if(isEncrypted) {
362             result = UtilsCryptography.decrypt(value, getSecretKey());
363         } else {
364             result = value;
365         }
366         return LOG.traceExit(result);
367     }

524     /**
525      * Copy input stream to output stream.
526      *
527      * @param inputStream input stream
528      * @param outputStream output stream
529      */
530     private void copyStream(final InputStream inputStream, final OutputStream outputStream) {
531         LOG.traceEntry("With an input and output stream");
532         final int bufferSize = 1024;
533         try {
534             final byte[] bytes = new byte[bufferSize];
535             for (;;) {
536                 final int count = inputStream.read(bytes, 0, bufferSize);
537                 if (count == -1) {
538                     break;
539                 }
540                 outputStream.write(bytes, 0, count);
541             }
542             inputStream.close();
543             outputStream.flush();
544             outputStream.close();
545         } catch (IOException ex) {
546             LOG.fatal("error when displaying the configuraiton file on the standard output", ex);
547         }
548         LOG.traceExit();
549     }
