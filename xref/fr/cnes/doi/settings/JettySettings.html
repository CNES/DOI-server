541     /**
542      * Add a parameter to a Jetty server.
543      *
544      * @param propName the name of the property to add
545      * @param propValue the value to set
546      * @throws IllegalArgumentException - if propValue is null
547      */
548     private void addParam(final String propName,
549             final Object propValue) {
550         LOG.traceEntry("Parameters : {} and {}", propName, propValue);
551         if (propValue == null) {
552             throw new IllegalArgumentException("propValue for " + propName + " is not defined");
553         }
554         final Parameter param = this.server.getContext().getParameters().getFirst(propName);
555         if (param == null) {
556             this.server.getContext().getParameters().add(propName, String.valueOf(propValue));
557             LOG.info("add to a new param {} : {}", propName, propValue);
558         } else {
559             param.setValue(String.valueOf(propValue));
560             LOG.info("add to an existant param {} : {}", propName, propValue);
561         }
562         LOG.traceExit();
563     }
