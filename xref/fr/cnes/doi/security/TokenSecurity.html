137     /**
138      * Creates a token.
139      *
140      * @param userID The user that creates the token
141      * @param projectID The project ID
142      * @param timeUnit The time unit for the date expiration
143      * @param amount the amount of timeUnit for the date expiration
144      * @return JWT token
145      * @throws fr.cnes.doi.exception.TokenSecurityException if the projectID is not first registered
146      */
147     public String generate(final String userID,
148             final int projectID,
149             final TokenSecurity.TimeUnit timeUnit,
150             final int amount) throws TokenSecurityException {
151         LOG.traceEntry("Parameters : {}, {}, {} and {}", userID, projectID, timeUnit, amount);
152         List<DOIProject> projects;
153         try {
154             projects = UniqueProjectName.getInstance().getProjects();
155         } catch (DOIDbException ex) {
156             projects = new ArrayList<>();
157         }
158         final Map<Integer, String> result = projects.stream().collect(
159                 Collectors.toMap(DOIProject::getSuffix, DOIProject::getProjectname));
160         final String projectName = result.get(projectID);
161         if (projectName.isEmpty()) {
162             throw LOG.throwing(new TokenSecurityException(
163                     Status.CLIENT_ERROR_BAD_REQUEST,
164                     "No register " + PROJECT_ID + ", please create one")
165             );
166         }
167         final Date now = Date.from(Instant.now());
168         final Date expirationTime = computeExpirationDate(now, timeUnit.getTimeUnit(), amount);
169 
170         final String token = Jwts.builder()
171                 .setIssuer(DoiSettings.getInstance().getString(Consts.APP_NAME))
172                 .setIssuedAt(Date.from(Instant.now()))
173                 .setSubject(userID)
174                 .claim(PROJECT_ID, projectID)
175                 .claim(PROJECT_NAME, projectName)
176                 .setExpiration(expirationTime)
177                 .signWith(
178                         SignatureAlgorithm.HS256,
179                         TextCodec.BASE64.decode(getTokenKey())
180                 )
181                 .compact();
182         LOG.debug(String.format("token generated : %s", token));
183         return LOG.traceExit(token);
184     }

187      * Creates a token. (no project ID required)
188      *
189      * @param userID The user that creates the token

249     /**
250      * Returns the token information.
251      *
252      * @param jwtToken token JWT
253      * @return the information or null when the token is expired.
254      * @throws DoiRuntimeException - if an error happens getting information from the token
255      */
256     public Jws<Claims> getTokenInformation(final String jwtToken) throws DoiRuntimeException {
257         LOG.traceEntry("Parameter : {}", jwtToken);
258         Jws<Claims> token;
259         try {
260             token = Jwts.parser()
261                     .requireIssuer(DoiSettings.getInstance().getString(Consts.APP_NAME))
262                     .setSigningKey(TextCodec.BASE64.decode(getTokenKey()))
263                     .parseClaimsJws(jwtToken);
264         } catch (UnsupportedJwtException
265                 | MalformedJwtException | SignatureException
266                 | IllegalArgumentException ex) {
267             throw LOG.throwing(new DoiRuntimeException("Unable to get the token information", ex));
268         } catch (ExpiredJwtException e) {
269             LOG.info("Cannot get the token information", e);
270             getTokenDB().deleteToken(jwtToken);
271             token = null;
272         }
273         return LOG.traceExit(token);
274     }

286     /**
287      * Computes the expiration date.
288      *
289      * @param now the current date
290      * @param calendarTime the unit time associated to the amount
291      * @param amount amount
292      * @return the expiration date
293      */
294     private Date computeExpirationDate(final Date now,
295             final int calendarTime,
296             final int amount) {
297         LOG.traceEntry("Parameters : {}, {} and {}", now, calendarTime, amount);
298         final Calendar calendar = Calendar.getInstance();
299         calendar.setTime(now);
300         calendar.add(calendarTime, amount);
301         return LOG.traceExit(calendar.getTime());
302     }

356         /**
357          * Returns the time unit from a value.
358          *
359          * @param value vale
360          * @return time unit
361          */
362         public static TimeUnit getTimeUnitFrom(final int value) {
363             TimeUnit result = null;
364             final TimeUnit[] units = TimeUnit.values();
365             for (final TimeUnit unit : units) {
366                 if (unit.getTimeUnit() == value) {
367                     result = unit;
368                     break;
369                 }
370             }
371             return result;
372         }
