98      /**
99       * Init users and groups in REALM.
100      */
101     private void initUsersGroups() {
102         LOG.traceEntry();
103 
104         // Add users
105         LOG.debug("Add users to REALM");
106         final AbstractUserRoleDBHelper manageUsers = PluginFactory.getUserManagement();
107         List<DOIUser> doiUsers;
108         try {
109             doiUsers = manageUsers.getUsers();
110         } catch (DOIDbException ex) {
111             doiUsers = new ArrayList<>();
112         }
113         final List<User> users = new ArrayList<>();
114         final List<User> admins = new ArrayList<>();
115         for (final DOIUser doiUser : doiUsers) {
116             final User user = new User(doiUser.getUsername());
117 
118             // Create list of user
119             users.add(user);
120 
121             // Create list of admin
122             if (doiUser.isAdmin()) {
123                 admins.add(user);
124             }
125         }
126         RoleAuthorizer.REALM.setUsers(users);
127         LOG.debug("List of users in realm : {}", RoleAuthorizer.REALM.getUsers());
128 
129         // Add Groups       
130         final Group administrators = new Group(GROUP_ADMIN, "Administrators");
131         LOG.debug("Add users to Administrators group");
132         administrators.getMemberUsers().addAll(admins);
133         LOG.debug("List of users in Administrators group {}", administrators.getMemberUsers());
134 
135         if (!RoleAuthorizer.REALM.getRootGroups().contains(administrators)) {
136             LOG.debug("Add administators group to rootGroups in REALM");
137             RoleAuthorizer.REALM.getRootGroups().add(administrators);
138         }
139         LOG.traceExit();
140     }

142     /**
143      * Sets Realm to Mds application.
144      *
145      * @param app Mds application
146      */
147     private void initForMds(final Application app) {
148         LOG.traceEntry(new JsonMessage(app));
149 
150         // we load projects from database
151         List<DOIProject> projects;
152         try {
153             projects = UniqueProjectName.getInstance().getProjects();
154         } catch (DOIDbException ex) {
155             projects = new ArrayList<>();
156         }
157         final AbstractUserRoleDBHelper manageUsers = PluginFactory.getUserManagement();
158         LOG.debug("{} projects have already been registered", projects.size());
159         for (final DOIProject project : projects) {
160             // for each project, create a role as the project name
161             final Integer projectID = project.getSuffix();
162             final Role role = new Role(app, String.valueOf(projectID), "Role " + String.valueOf(
163                     projectID) + " for " + app.getName());
164 
165             // get the users for a role.
166             List<DOIUser> doiUsers;
167             try {
168                 doiUsers = manageUsers.getUsersFromRole(projectID);
169             } catch (DOIDbException ex) {
170                 doiUsers = new ArrayList<>();
171             }
172             final List<User> usersFromProject = new ArrayList<>();
173             for (final DOIUser doiUser : doiUsers) {
174                 final User user = REALM.findUser(doiUser.getUsername());
175                 if (user != null) {
176                     usersFromProject.add(user);
177                 }
178             }
179 
180             // create a role authorizer for each user related to a project
181             for (final User user : usersFromProject) {
182                 LOG.debug("Add user {} to role {} for {}", user, projectID, app.getName());
183                 RoleAuthorizer.REALM.map(user, role);
184             }
185         }
186 
187         app.getContext().setDefaultEnroler(RoleAuthorizer.REALM.getEnroler());
188         app.getContext().setDefaultVerifier(RoleAuthorizer.REALM.getVerifier());
189 
190         LOG.traceExit();
191     }

193     /**
194      * Finds a group by its name
195      *
196      * @param name group name
197      * @return the group
198      */
199     private Group findGroupByName(final String name) {
200         LOG.traceEntry("Parameter : {}", name);
201         final List<Group> groups = RoleAuthorizer.REALM.getRootGroups();
202         Group searchedGroup = null;
203         for (final Group group : groups) {
204             LOG.debug("group name : {}", group.getName());
205             if (group.getName().equals(name)) {
206                 searchedGroup = group;
207                 LOG.debug("group found");
208                 break;
209             }
210         }
211         if (searchedGroup == null) {
212             LOG.error("Please, create a group {}", name);
213             throw LOG.throwing(new DoiRuntimeException("Please, create a group " + name));
214         }
215         return LOG.traceExit(searchedGroup);
216     }

234     /**
235      * Init Realm for an application.
236      *
237      * @param app application
238      * @return True when the realm is initialized otherwise False
239      */
240     public boolean createRealmFor(final Application app) {
241         LOG.traceEntry(new JsonMessage(app));
242 
243         boolean isCreated;
244         switch (app.getName()) {
245             case DoiMdsApplication.NAME:
246                 initForMds(app);
247                 isCreated = true;
248                 LOG.debug("Init for MDS ... done");
249                 break;
250             case AdminApplication.NAME:
251                 initForAdmin(app);
252                 isCreated = true;
253                 LOG.debug("Init for admin ... done");
254                 break;
255             default:
256                 LOG.debug("No Realm is initialized for this application {}", app.getName());
257                 isCreated = false;
258                 break;
259         }
260 
261         return LOG.traceExit(isCreated);
262     }

264     /**
265      * Adds/removes the <i>users</i> group to the new role name related to the application MDS
266      *
267      * @param obs observable
268      * @param obj message
269      */
270     @Requirement(reqId = Requirement.DOI_SRV_130, reqName = Requirement.DOI_SRV_130_NAME)
271     @Requirement(reqId = Requirement.DOI_INTER_030, reqName = Requirement.DOI_INTER_030_NAME)
272     @Override
273     public void update(final Observable obs,
274             final Object obj) {
275         LOG.traceEntry(new JsonMessage(obs));
276         LOG.traceEntry(new JsonMessage(obj));
277 
278         // Loads the admin group - admin group is defined by default
279         final Group adminGroup = findGroupByName(GROUP_ADMIN);
280 
281         // Loads the application MDS related to admin group
282         final Application mds = loadApplicationBy(adminGroup, DoiMdsApplication.NAME);
283 
284         if (mds == null) {
285             LOG.info("{} is not defined in the REALM", DoiMdsApplication.NAME);
286         } else {
287             updateObserver(obs, obj, mds);
288         }
289         LOG.traceExit();
290     }

292     /**
293      * Updates REALM for mds.
294      *
295      * @param obs observable
296      * @param obj message
297      * @param mds application
298      */
299     private void updateObserver(final Observable obs,
300             final Object obj,
301             final Application mds) {
302         LOG.traceEntry(new JsonMessage(obs));
303         LOG.traceEntry(new JsonMessage(obj));
304         LOG.traceEntry(new JsonMessage(mds));
305 
306         if (obs instanceof AbstractProjectSuffixDBHelper) {
307             LOG.debug("Observable is a ProjectSuffixDB type");
308 
309             final String[] message = (String[]) obj;
310 
311             final String operation = message[0];
312             final String roleName = message[1];
313             final AbstractUserRoleDBHelper manageUsers = PluginFactory.getUserManagement();
314             List<DOIUser> doiUsers;
315             try {
316                 doiUsers = manageUsers.getUsersFromRole(Integer.parseInt(roleName));
317             } catch (DOIDbException ex) {
318                 doiUsers = new ArrayList<>();
319             }
320             final List<User> usersFromProject = new ArrayList<>();
321             for (final DOIUser doiUser : doiUsers) {
322                 final User user = new User(doiUser.getUsername());
323                 usersFromProject.add(user);
324             }
325             final List<User> users = usersFromProject;
326             switch (operation) {
327                 case AbstractProjectSuffixDBHelper.ADD_RECORD:
328                     for (final User user : users) {
329                         RoleAuthorizer.REALM.map(user, Role.get(mds, roleName));
330                         LOG.debug("Adds the user {} to the new role {} related to the {}", user,
331                                 roleName, mds.getName());
332                     }
333                     break;
334                 case AbstractProjectSuffixDBHelper.DELETE_RECORD:
335                     for (final User user : users) {
336                         RoleAuthorizer.REALM.unmap(user, mds, roleName);
337                         LOG.info("Remove the user {} to the role {} related to the {}", user,
338                                 roleName, mds.getName());
339                     }
340                     break;
341                 default:
342                     LOG.error("operation {} was not expected", operation);
343                     break;
344             }
345         }
346         LOG.traceExit();
347     }

349     /**
350      * Loads the application related to a group with a name
351      *
352      * @param group group linked to an application
353      * @param appName application name
354      * @return the application or null if the application is not defined in the REALM
355      */
356     private Application loadApplicationBy(final Group group,
357             final String appName) {
358         LOG.traceEntry("Parameters : {} and {}", group, appName);
359 
360         final Set<Role> roles = RoleAuthorizer.REALM.findRoles(group);
361         final Iterator<Role> roleIter = roles.iterator();
362         Application searchedApp = null;
363         boolean isFound = false;
364         while (roleIter.hasNext() && !isFound) {
365             final Role role = roleIter.next();
366             final Application app = role.getApplication();
367             if (app.getName().equals(appName)) {
368                 searchedApp = app;
369                 isFound = true;
370                 break;
371             }
372         }
373 
374         return LOG.traceExit(searchedApp);
375     }
