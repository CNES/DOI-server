86      /**
87       * Returns the keys related to a value within a map.
88       *
89       * @param <T> the type of the key
90       * @param <E> the type of the value
91       * @param map map
92       * @param value value to search
93       * @return the keys related to a value
94       */
95      public static <T, E> Set<T> getKeysByValue(final Map<T, E> map, final E value) {
96          final Set<T> keys = new HashSet<>();
97          for (final Entry<T, E> entry : map.entrySet()) {
98              final E entryVal = entry.getValue();
99              if (entryVal.equals(value)) {
100                 keys.add(entry.getKey());
101             }
102         }
103         return keys;
104     }

106     /**
107      * <p>
108      * Defensive programming technique to change a {@code null} reference to an empty one.
109      *
110      * <p>
111      * This method returns an empty array for a {@code null} input array.
112      *
113      * @param array the array to check for {@code null} or empty
114      * @param type the class representation of the desired array
115      * @param <T> the class type
116      * @return the same array, {@code public static} empty array if {@code null}
117      * @throws IllegalArgumentException if the type argument is null
118      */
119     public static <T> T[] nullToEmpty(final T[] array,
120             final Class<T[]> type)
121             throws IllegalArgumentException {
122         if (type == null) {
123             throw new IllegalArgumentException("The type must not be null");
124         }
125 
126         T[] result;
127         if (array == null) {
128             result = type.cast(Array.newInstance(type.getComponentType(), 0));
129         } else {
130             result = array;
131         }
132         return result;
133     }

135     /**
136      * <p>
137      * Defensive programming technique to change a {@code null} reference to an empty one.
138      *
139      * <p>
140      * This method returns an empty array for a {@code null} input array.
141      *
142      * <p>
143      * As a memory optimizing technique an empty array passed in will be overridden with the empty
144      * {@code public static} references in this class.
145      *
146      * @param array the array to check for {@code null} or empty
147      * @return the same array, {@code public static} empty array if {@code null} or empty input
148      */
149     public static Object[] nullToEmpty(final Object[] array) {
150         Object[] result;
151         if (isEmpty(array)) {
152             result = EMPTY_OBJECT_ARRAY;
153         } else {
154             result = array;
155         }
156         return result;
157     }

170     /**
171      * <p>
172      * Returns the length of the specified array. This method can deal with {@code Object} arrays
173      * and with primitive arrays.
174      *
175      * <p>
176      * If the input array is {@code null}, {@code 0} is returned.
177      *
178      * <pre>
179      * Utils.getLength(null)            = 0
180      * Utils.getLength([])              = 0
181      * Utils.getLength([null])          = 1
182      * Utils.getLength([true, false])   = 2
183      * Utils.getLength([1, 2, 3])       = 3
184      * Utils.getLength(["a", "b", "c"]) = 3
185      * </pre>
186      *
187      * @param array the array to retrieve the length from, may be null
188      * @return The length of the array, or {@code 0} if the array is {@code null}
189      * @throws IllegalArgumentException if the object argument is not an array.
190      */
191     public static int getLength(final Object array) throws IllegalArgumentException {
192         int length;
193         if (array == null) {
194             length = 0;
195         } else {
196             length = Array.getLength(array);
197         }
198         return length;
199     }

201     /**
202      * Transform password to stars
203      *
204      * @param password password to transform
205      * @return stars
206      */
207     public static String transformPasswordToStars(final String password) {
208         final StringBuilder passStars = new StringBuilder();
209         for (int i = 0; i < password.length(); i++) {
210             passStars.append("*");
211         }
212         return passStars.toString();
