114     /**
115      * Stops the server
116      *
117      * @param server HTTP or HTTPS server
118      */
119     private static void stopServer(final Thread server) {
120         LOG.traceEntry();
121         try {
122             try {
123                 doiServer.stop();
124             } catch (Exception ex) {
125                 LOG.fatal("Unable to stop the server", ex);
126             } finally {
127                 LOG.info("Interrups the server, which is stopping");
128                 server.interrupt();
129                 server.join();
130             }
131         } catch (InterruptedException e) {
132             LOG.fatal("Cannot interrupt the server", e);
133         }
134         LOG.traceExit();
135     }

144         final DoiSettings settings = DoiSettings.getInstance();
145         final String progName = settings.getString(Consts.APP_NAME) + "-" + settings.getString(
146                 Consts.VERSION) + ".jar";
147         final String stopPid = getCurrentPid(progName);
148         if (stopPid == null) {
149             try {
150                 server.start();
151                 infoProject();
152             } catch (Exception ex) {
153                 LOG.info("Unable to start the server");
154             }
155         } else {
156             LOG.info("The server is already started");
157         }
158     }

212     /**
213      * Launches the server.
214      *
215      * @param settings Configuration
216      */
217     private static void launchServer(final DoiSettings settings) {
218         LOG.traceEntry();
219         try {
220             settings.validConfigurationFile();
221             doiServer = new DoiServer(settings);
222             final Thread server = new Thread() {
223                 @Override
224                 public void run() {
225                     startServer(doiServer);
226                 }
227             };
228 
229             Runtime.getRuntime().addShutdownHook(new Thread() {
230                 @Override
231                 public void run() {
232                     LOG.info("interrupt received, killing serverâ€¦");
233                     stopServer(server);
234                 }
235             });
236 
237             server.start();
238         } catch (ClientMdsException| DoiRuntimeException ex) {
239             LOG.info("Error when starting the server: " + ex.getMessage());
240         } 
241         LOG.traceExit();
242     }

257     /**
258      * Return the server PID from the system.
259      *
260      * @param serverName server name
261      * @return the PID or null
262      */
263     private static String getCurrentPid(final String serverName) {
264         LOG.traceEntry("Parameter\n\tserverName:{}", serverName);
265         String stopPid = null;
266         BufferedReader reader = null;
267         Process pro = null;
268         try {
269             pro = Runtime.getRuntime().exec("ps aux");
270             reader = new BufferedReader(new InputStreamReader(pro.
271                     getInputStream(), Charset.defaultCharset()));
272             final String processName = java.lang.management.ManagementFactory.getRuntimeMXBean().
273                     getName();
274             final String myPid = processName.split("@")[0];
275             String line;
276             while ((line = reader.readLine()) != null) {
277                 if (line.contains(serverName) && line.contains("java")) {
278                     final String[] split = line.split("\\s+");
279                     final String currentPid = split[1].trim();
280                     if (!currentPid.equals(myPid)) {
281                         stopPid = currentPid;
282                         break;
283                     }
284                 }
285             }
286         } catch (IOException ex) {
287         } finally {
288             if (reader != null) {
289                 try {
290                     reader.close();
291                 } catch (IOException ex) {
292                 }
293             }
294             if (pro != null) {
295                 pro.destroy();
296             }
297         }
298         return LOG.traceExit(stopPid);
299     }

301     /**
302      * Main.
303      *
304      * @param argv command line arguments
305      */
306     public static void main(final String[] argv) {
307         final DoiSettings settings = DoiSettings.getInstance();
308         final String progName = settings.getString(Consts.APP_NAME) + "-" + settings.getString(
309                 Consts.VERSION) + ".jar";
310         final String progNameWithJar = "java -jar " + progName;
311 
312         int c;
313         String arg;
314 
315         final StringBuffer sb = new StringBuffer();
316         final LongOpt[] longopts = new LongOpt[10];
317         longopts[0] = new LongOpt("help", LongOpt.NO_ARGUMENT, null, 'h');
318         longopts[1] = new LongOpt("version", LongOpt.NO_ARGUMENT, null, 'v');
319         longopts[2] = new LongOpt("secret", LongOpt.REQUIRED_ARGUMENT, sb, 0);
320         longopts[3] = new LongOpt("decryptProperties", LongOpt.REQUIRED_ARGUMENT, null, 'z');
321         longopts[4] = new LongOpt("cryptProperties", LongOpt.REQUIRED_ARGUMENT, null, 'y');
322         longopts[5] = new LongOpt("key-sign", LongOpt.NO_ARGUMENT, null, 'k');
323         longopts[6] = new LongOpt("key-sign-secret", LongOpt.REQUIRED_ARGUMENT, null, 'a');
324         longopts[7] = new LongOpt("start", LongOpt.NO_ARGUMENT, null, 's');
325         longopts[8] = new LongOpt("stop", LongOpt.NO_ARGUMENT, null, 't');
326         longopts[8] = new LongOpt("status", LongOpt.NO_ARGUMENT, null, 'l');
327         //
328         final Getopt g = new Getopt(progNameWithJar, argv, "hvdstke:c:f:y:z:a:b:", longopts);
329         //
330         while ((c = g.getopt()) != -1) {
331             switch (c) {
332                 case 0:
333                     final String secretKey = g.getOptarg();
334                     if (secretKey.length() != BITS_16) {
335                         throw new IllegalArgumentException(
336                                 "The secret key must have 16 characters.");
337                     } else {
338                         settings.setSecretKey(secretKey);
339                     }
340                     break;
341                 //
342                 case 'a':
343                     LOG.debug("a option is selected");
344                     final String secretSignToken = g.getOptarg();
345                     TokenSecurity.getInstance().setTokenKey(secretSignToken);
346                     break;
347                 case 'h':
348                     LOG.debug("h option is selected");
349                     displayHelp();
350                     break;
351                 //
352                 case 's':
353                     LOG.debug("s option is selected");
354                     launchServer(settings);
355                     break;
356                 case 't':
357                     LOG.debug("t option is selected");
358                     stopServer();
359                     break;
360                 case 'l':
361                     LOG.debug("l option is selected");
362                     statusServer();
363                     break;
364                 //
365                 case 'k':
366                     LOG.debug("k option is selected");
367                     LOG.info(TokenSecurity.createKeySignatureHS256());
368                     break;
369                 //
370                 case 'e':
371                     LOG.debug("e option is selected");
372                     arg = g.getOptarg();
373                     try {
374                         LOG.info(UtilsCryptography.decrypt(arg, settings.getSecretKey()));
375                     } catch (DoiRuntimeException ex) {
376                         LOG.fatal("Unable to decrypt {} : {}", arg, ex.getMessage());
377                     }
378                     break;
379                 //
380                 case 'c':
381                     LOG.debug("c option is selected");
382                     arg = g.getOptarg();
383                     try {
384                         LOG.info(UtilsCryptography.encrypt(arg, settings.getSecretKey()));
385                     } catch (DoiRuntimeException ex) {
386                         LOG.fatal("Unable to encrypt {} : {}", arg, ex.getMessage());
387                     }
388                     break;
389                 //
390                 case 'd':
391                     LOG.debug("d option is selected");
392                     settings.displayConfigFile();
393                     break;
394                 //
395                 case 'f':
396                     LOG.debug("f option is selected");
397                     try {
398                         settings.setPropertiesFile(g.getOptarg());
399                     } catch (IOException ex) {
400                         LOG.fatal(ex.getMessage());
401                     }
402                     break;
403                 //
404                 case 'v':
405                     LOG.debug("v option is selected");
406                     displayVersion();
407                     break;
408                 //
409                 case 'y':
410                     LOG.debug("y option is selected");
411                     try {
412                         final byte[] encodedFile = Files.readAllBytes(Paths.get(g.getOptarg()));
413                         String contentFile = new String(encodedFile, StandardCharsets.UTF_8);
414                         contentFile = UtilsCryptography.encrypt(contentFile,
415                                 settings.getSecretKey());
416                         LOG.info(contentFile);
417                     } catch (IOException ex) {
418                         LOG.fatal("Error: {}", ex.getMessage());
419                     }
420                     break;
421                 //
422                 case 'z':
423                     LOG.debug("z option is selected");
424                     InputStream inputStream = null;
425                     BufferedReader reader = null;
426                     Reader inputReader = null;
427 
428                     try {
429                         inputStream = new FileInputStream(g.getOptarg());
430                         inputReader = new InputStreamReader(inputStream, StandardCharsets.UTF_8);
431                         reader = new BufferedReader(inputReader);
432                         String content = reader.lines().collect(Collectors.joining("\n"));
433                         content = UtilsCryptography.decrypt(content, settings.getSecretKey());
434                         LOG.info(content);
435                     } catch (FileNotFoundException ex) {
436                         LOG.fatal("Error: {}", ex.getMessage());
437                     } finally {
438                         if (inputStream != null) {
439                             try {
440                                 inputStream.close();
441                             } catch (IOException e) {
442                                 LOG.fatal("Error closing inputstream: {}", e.getMessage(), e);
443                             }
444                         }
445                         if (inputReader != null) {
446                             try {
447                                 inputReader.close();
448                             } catch (IOException ex) {
449                                 LOG.fatal("Error closing inputReader: {}", ex.getMessage(), ex);
450                             }
451                         }
452                         if (reader != null) {
453                             try {
454                                 reader.close();
455                             } catch (IOException ex) {
456                                 LOG.fatal("Error closing reader: {}", ex.getMessage(), ex);
457                             }
458                         }
459                     }
460                     break;
461                 case '?':
462                     break; // getopt() already printed an error
463                 //
464                 default:
465                     LOG.debug("getopt() returned {}\n", c);
466             }
467         }
468         //
469         for (int i = g.getOptind(); i < argv.length; i++) {
470             LOG.info("Non option argv element: {}\n", argv[i]);
471         }
472 
473         if (argv.length == 0) {
474             displayHelp();
475         }
476 
477     }
