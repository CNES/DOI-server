219     /**
220      * Configures the Server in HTTP and HTTPS.
221      */
222     @Requirement(reqId = Requirement.DOI_ARCHI_010, reqName = Requirement.DOI_ARCHI_010_NAME)
223     private void configureServer() {
224         LOG.traceEntry();
225         final boolean isHttpStarted = initHttpServer();
226         final boolean isHttpsStarted = initHttpsServer();
227         if (isHttpStarted || isHttpsStarted) {
228             initClients();
229             initAttachApplication();
230         } else {
231             LOG.warn("No server is configured, please check your configuration file");
232         }
233 
234         LOG.traceExit();
235     }

237     /**
238      * Inits the HTTP server.
239      *
240      * @return True when the Http server is configured to start.
241      */
242     private boolean initHttpServer() {
243         LOG.traceEntry();
244         final boolean isConfigured;
245         if (settings.hasValue(Consts.SERVER_HTTP_PORT)) {
246             final String httpPort = settings.getString(Consts.SERVER_HTTP_PORT);
247             final Server serverHttp = startHttpServer(Integer.parseInt(httpPort));
248             this.getServers().add(serverHttp);
249             initJettyConfiguration(serverHttp);
250             isConfigured = true;
251         } else {
252             isConfigured = false;
253         }
254         return LOG.traceExit(isConfigured);
255     }

301     /**
302      * Routes the applications.
303      */
304     private void initAttachApplication() {
305         LOG.traceEntry();
306         final Application appDoiProject = new DoiMdsApplication();
307         final Application appAdmin = new AdminApplication();
308         this.getDefaultHost().attach(MDS_URI, appDoiProject);
309         this.getDefaultHost().attach(CITATION_URI, new DoiCrossCiteApplication());
310         this.getDefaultHost().attachDefault(appAdmin);
311         // Set authentication 
312         RoleAuthorizer.getInstance().createRealmFor(appDoiProject);
313         RoleAuthorizer.getInstance().createRealmFor(appAdmin);
314         // Set authentication user as admin
315         final String doiAdmin = PluginFactory.getAuthenticationSystem().getDOIAdmin();
316         addAuthenticationUserAsAdmin(doiAdmin);
317         LOG.traceExit();
318     }

320     /**
321      * Adds an authentication system user as administrator of the DOI server
322      *
323      * @param username username
324      */
325     private void addAuthenticationUserAsAdmin(final String username) {
326         LOG.traceEntry("Parameter\n   username: {}", username);
327         final IAuthenticationDBHelper authenticationService = PluginFactory.
328                 getAuthenticationSystem();
329         final AbstractUserRoleDBHelper manageUsers = PluginFactory.getUserManagement();
330         try {
331             boolean isFound = false;
332             final List<AuthSystemUser> authenticationUsers = authenticationService.
333                     getDOIProjectMembers();
334             for (final AuthSystemUser authenticationUser : authenticationUsers) {
335                 if (authenticationUser.getUsername().equals(username)) {
336                     manageUsers.setUserToAdminGroup(authenticationUser.getUsername());
337                     isFound = true;
338                     break;
339                 }
340             }
341             if (!isFound) {
342                 LOG.warn("{} is not registered in the authentication system - Cannot create "
343                         + "the administrator in DOI database",
344                         username
345                 );
346             }
347         } catch (AuthenticationAccessException ex) {
348             LOG.catching(ex);
349             LOG.warn("Cannot create an administrator: {}", ex);
350         }
351 
352         LOG.traceExit();
353     }

381     /**
382      * Creates a HTTPS server
383      *
384      * @param port HTTPS port
385      * @return the HTTPS server
386      */
387     private Server startHttpsServer(final Integer port) {
388         LOG.traceEntry(MESSAGE_TPL, PARAMETERS, port);
389         final String pathKeyStore;
390         if (settings.hasValue(Consts.SERVER_HTTPS_KEYSTORE_PATH)) {
391             pathKeyStore = settings.getString(Consts.SERVER_HTTPS_KEYSTORE_PATH);
392             LOG.debug("path key store value loaded from a custom configuration file");
393         } else {
394             pathKeyStore = extractKeyStoreToPath();
395             LOG.debug("path key store value loaded from an internal configuration");
396         }
397 
398         final String pathKeyTrustStore;
399         if (settings.hasValue(Consts.SERVER_HTTPS_TRUST_STORE_PATH)) {
400             pathKeyTrustStore = settings.getString(Consts.SERVER_HTTPS_TRUST_STORE_PATH);
401         } else {
402             pathKeyTrustStore = extractKeyStoreToPath();
403         }
404 
405         // create embedding https jetty server
406         final Server server = new Server(new Context(), Protocol.HTTPS, port, this);
407         final Series<Parameter> parameters = server.getContext().getParameters();
408 
409         LOG.debug(MESSAGE_TPL, RESTLET_MAX_TOTAL_CONNECTIONS, DoiSettings.getInstance().getString(
410                 fr.cnes.doi.settings.Consts.RESTLET_MAX_TOTAL_CONNECTIONS,
411                 DEFAULT_MAX_TOTAL_CONNECTIONS));
412         parameters.set(RESTLET_MAX_TOTAL_CONNECTIONS, DoiSettings.getInstance().getString(
413                 fr.cnes.doi.settings.Consts.RESTLET_MAX_TOTAL_CONNECTIONS,
414                 DEFAULT_MAX_TOTAL_CONNECTIONS));
415 
416         LOG.debug(MESSAGE_TPL, RESTLET_MAX_CONNECTIONS_PER_HOST, DoiSettings.getInstance().
417                 getString(fr.cnes.doi.settings.Consts.RESTLET_MAX_CONNECTIONS_PER_HOST,
418                         DEFAULT_MAX_CONNECTIONS_PER_HOST));
419         parameters.set(RESTLET_MAX_CONNECTIONS_PER_HOST, DoiSettings.getInstance().getString(
420                 fr.cnes.doi.settings.Consts.RESTLET_MAX_CONNECTIONS_PER_HOST,
421                 DEFAULT_MAX_CONNECTIONS_PER_HOST));
422 
423         LOG.debug(MESSAGE_TPL, SSL_CTX_FACTORY, DEFAULT_SSL_CTX);
424         parameters.add(SSL_CTX_FACTORY, DEFAULT_SSL_CTX);
425 
426         // Specifies the path for the keystore used by the server
427         LOG.debug(MESSAGE_TPL, KEY_STORE_PATH, pathKeyStore);
428         parameters.add(KEY_STORE_PATH, pathKeyStore);
429 
430         // Specifies the password for the keystore containing several keys
431         LOG.debug(MESSAGE_TPL, KEY_STORE_PWD, "xxxxxxx");
432         parameters.add(KEY_STORE_PWD, settings.getSecret(Consts.SERVER_HTTPS_KEYSTORE_PASSWD));
433 
434         // Specifies the type of the keystore
435         LOG.debug(MESSAGE_TPL, KEY_STORE_TYPE, KeyStore.getDefaultType());
436         parameters.add(KEY_STORE_TYPE, KeyStore.getDefaultType());
437 
438         // Specifies the password of the specific key used
439         LOG.debug(MESSAGE_TPL, KEY_PWD, "xxxxxxxx");
440         parameters.add(KEY_PWD, settings.getSecret(Consts.SERVER_HTTPS_SECRET_KEY));
441 
442         // Specifies the path to the truststore
443         LOG.debug(MESSAGE_TPL, TRUST_STORE_PATH, pathKeyTrustStore);
444         parameters.add(TRUST_STORE_PATH, pathKeyTrustStore);
445 
446         // Specifies the password of the truststore
447         LOG.debug(MESSAGE_TPL, TRUST_STORE_PWD, "xxxxx");
448         parameters.add(TRUST_STORE_PWD, settings.getSecret(Consts.SERVER_HTTPS_TRUST_STORE_PASSWD));
449 
450         // Specifies the type of the truststore
451         LOG.debug(MESSAGE_TPL, TRUST_STORE_TYPE, KeyStore.getDefaultType());
452         parameters.add(TRUST_STORE_TYPE, KeyStore.getDefaultType());
453 
454         return LOG.traceExit(server);
455     }

457     /**
458      * Extracts keystore for JAR and copy it in a directory in order to use it.
459      *
460      * @return the path of the new location of the keystore.
461      */
462     private String extractKeyStoreToPath() {
463         LOG.traceEntry();
464         String result;
465         final Representation jks = new ClientResource(
466                 LocalReference.createClapReference("class/" + JKS_FILE)
467         ).get();
468         try {
469             final Path outputDirectory = new File(JKS_DIRECTORY).toPath();
470             if (Files.notExists(outputDirectory)) {
471                 Files.createDirectory(outputDirectory);
472                 LOG.info("Creates {} directory to extract {} in it", outputDirectory, JKS_FILE);
473             }
474             final String outputJksLocation = outputDirectory.getFileName()
475                     + File.separator
476                     + JKS_FILE;
477 
478             final File outputFile = new File(outputJksLocation);
479             Files.copy(jks.getStream(),
480                     outputFile.toPath(),
481                     java.nio.file.StandardCopyOption.REPLACE_EXISTING
482             );
483             LOG.info("Copy/replace if exists {} in {}", JKS_FILE, outputDirectory, JKS_FILE);
484             result = outputJksLocation;
485         } catch (IOException ex) {
486             LOG.fatal("Unable to extract keystore from class/" + JKS_FILE, ex);
487             result = "";
488         }
489         return LOG.traceExit(result);
490     }

508         LOG.info("Stopping the server ...");
509         super.stop();

511         PluginFactory.getAuthenticationSystem().release();
512         LOG.info("Stopping Project plugin");
513         PluginFactory.getProjectSuffix().release();
514         LOG.info("Stopping Token plugin");
515         PluginFactory.getToken().release();
516         LOG.info("Stopping UserManagement plugin");
517         PluginFactory.getUserManagement().release();
518         EmailSettings.getInstance().sendMessage("[DOI] Stopping Server",
519                 "Ther server has been interrupted");
520         LOG.info("Server stopped");
521     }
