19  package fr.cnes.doi.services;
20  
21  import fr.cnes.doi.db.AbstractUserRoleDBHelper;
22  import fr.cnes.doi.db.IAuthenticationDBHelper;
23  import java.util.List;
24  
25  import fr.cnes.doi.exception.AuthenticationAccessException;
26  import fr.cnes.doi.db.model.AuthSystemUser;
27  import fr.cnes.doi.exception.DOIDbException;
28  import fr.cnes.doi.db.model.DOIUser;
29  import org.apache.logging.log4j.LogManager;
30  import org.apache.logging.log4j.Logger;
31  import fr.cnes.doi.plugin.PluginFactory;
32  
33  /**
34   * Updates the database from the authentication system.
35   *
36   * @author Jean-Christophe Malapert (jean-christophe.malapert@cnes.fr)
37   */
38  public class DOIUsersUpdate implements Runnable {
39  
40      /**
41       * Logger.
42       */
43      private static final Logger LOG = LogManager.getLogger(DOIUsersUpdate.class.getName());
44  
45      /**
46       * Access to the authentication system.
47       */
48      private static final IAuthenticationDBHelper AUTHENTICATION_SERVICE = PluginFactory.
49              getAuthenticationSystem();
50  
51      /**
52       * Fills the DOI users database from the members of the authentication system.
53       *
54       * @throws AuthenticationAccessException When an authentication problem occurs
55       * @throws DOIDbException When a SQL problem occurs
56       */
57      private void updateDoiServerDataBaseFromAuthSystem() throws AuthenticationAccessException,
58              DOIDbException {
59          LOG.traceEntry();
60          final AbstractUserRoleDBHelper manageUsers = PluginFactory.getUserManagement();
61          final List<AuthSystemUser> authMembers = AUTHENTICATION_SERVICE.getDOIProjectMembers();
62          LOG.debug("Authentication system members: {}", authMembers);
63          final List<DOIUser> dbusers = manageUsers.getUsers();
64          LOG.debug("Users from database: {}", dbusers);
65          LOG.debug("remove from database, users that are no longer members of doi_server project");
66          for (final DOIUser dbuser : dbusers) {
67              if (!isContained(dbuser, authMembers)) {
68                  LOG.debug("User {} is removed from database", dbuser.getUsername());
69                  manageUsers.removeDOIUser(dbuser.getUsername());
70              }
71          }
72          LOG.debug("add to database users that are new members of doi_server project");
73          for (final AuthSystemUser authMember : authMembers) {
74              if (!isContained(authMember, dbusers)) {
75                  // add authMember to doi database
76                  LOG.debug("authSystemUser {} is added to database as simple user", authMember.
77                          getUsername());
78                  manageUsers.addDOIUser(authMember.getUsername(), false, authMember.getEmail());
79              }
80          }
81          LOG.traceExit();
82      }
83  
84      /**
85       * Tests if the authMember of the authentication mechanism is contained in the users from the
86       * database.
87       *
88       * @param authMember authMember of the authentication mechanism
89       * @param dbusers users database
90       * @return True when the authMember is contained in the users from database otherwise false
91       */
92      private boolean isContained(final AuthSystemUser authMember, final List<DOIUser> dbusers) {
93          LOG.traceEntry("Parameters {},", authMember, dbusers);
94          boolean isContained = false;
95          for (final DOIUser dbuser : dbusers) {
96              if (dbuser.getUsername().equals(authMember.getUsername())) {
97                  isContained = true;
98                  break;
99              }
100         }
101         return LOG.traceExit(isContained);
102     }
103 
104     /**
105      * Tests if a user from the database is contained in the members from the authentication system.
106      *
107      * @param dbuser the user from the database
108      * @param authMembers the users from the authentication system
109      * @return True when the user from the database is contained from the users from the
110      * authentication system
111      */
112     private boolean isContained(final DOIUser dbuser, final List<AuthSystemUser> authMembers) {
113         LOG.traceEntry("Parameters {},", dbuser, authMembers);
114         boolean isContained = false;
115         for (final AuthSystemUser authenticationUser : authMembers) {
116             if (authenticationUser.getUsername().equals(dbuser.getUsername())) {
117                 isContained = true;
118                 break;
119             }
120         }
121         return LOG.traceExit(isContained);
122     }
123 
124     /**
125      * {@inheritDoc}
126      */
127     @Override
128     public void run() {
129         LOG.info("executing task that updates database from the authentication system !");
130         try {
131             this.updateDoiServerDataBaseFromAuthSystem();
132         } catch (AuthenticationAccessException | DOIDbException e) {
133             LOG.error("error occured when calling DOIUsersUpdate job", e);
134         }
135     }
