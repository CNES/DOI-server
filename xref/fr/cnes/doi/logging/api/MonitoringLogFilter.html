19  package fr.cnes.doi.logging.api;

21  import fr.cnes.doi.services.DoiMonitoring;
22  import fr.cnes.doi.settings.Consts;
23  import fr.cnes.doi.settings.DoiSettings;
24  import fr.cnes.doi.settings.EmailSettings;
25  import fr.cnes.doi.utils.Utils;
26  import fr.cnes.doi.utils.spec.Requirement;
27  import org.apache.logging.log4j.LogManager;
28  import org.restlet.Context;
29  import org.restlet.Request;
30  import org.restlet.Response;
31  import org.restlet.data.Method;
32  import org.restlet.engine.log.LogFilter;
33  import org.restlet.service.LogService;

35  /**
36   * Filter to monitor the applications speed performance.
37   *
38   * @author Jean-Christophe Malapert (jean-christophe.malapert@cnes.fr)
39   */
40  @Requirement(reqId = Requirement.DOI_MONIT_010, reqName = Requirement.DOI_MONIT_010_NAME)
41  @Requirement(reqId = Requirement.DOI_ARCHI_020, reqName = Requirement.DOI_ARCHI_020_NAME)
42  public class MonitoringLogFilter extends LogFilter {
43  
44      /**
45       * Threshold from which an alarm is send.
46       */
47      private static final float THRESHOLD_SPEED_PERCENT = Float
48              .valueOf(DoiSettings.getInstance().getString(Consts.THRESHOLD_SPEED_PERCENT));
49  
50      /**
51       * The monitoring object
52       */
53      private final DoiMonitoring monitoring;
54  
55      /**
56       * Constructs a filter that filters applications to monitor
57       *
58       * @param context the Context
59       * @param doiMonitoring DOI monitoring
60       * @param logService the {@link LogService}
61       */
62      public MonitoringLogFilter(final Context context, final DoiMonitoring doiMonitoring,
63              final LogService logService) {
64          super(context, logService);
65          this.monitoring = doiMonitoring;
66          this.logService
67                  .setResponseLogFormat(DoiSettings.getInstance().getString(Consts.LOG_FORMAT));
68      }
69  
70      /**
71       * Allows filtering after processing by the next Restlet. Does nothing by
72       * default.
73       *
74       * @param request request
75       * @param response response
76       */
77      @Override
78      protected void afterHandle(final Request request, final Response response) {
79          if (this.logService.isLoggable(request)) {
80  
81              LogManager.getLogger(Utils.HTTP_LOGGER_NAME)
82                      .info(this.logService.getResponseLogMessage(response, 1));
83          }
84          if (response.getStatus().isSuccess()) {
85              final String path = request.getResourceRef().getPath();
86              final Method method = request.getMethod();
87              final long startTime = (Long) request.getAttributes().get("org.restlet.startTime");
88              final int duration = (int) (System.currentTimeMillis() - startTime);
89              if (monitoring.isRegistered(method, path)) {
90                  monitoring.addMeasurement(method, path, duration);
91                  final String description = monitoring.getDescription(method, path);
92                  final String methodName = method.getName();
93                  final float average = monitoring.getCurrentAverage(method, path);
94                  LogManager.getLogger(Utils.APP_LOGGER_NAME).info(
95                          "{}({} {}) - current speed average : {} ms - "
96                          + "current measure: {} ms",
97                          description, methodName, path, average, duration);
98                  sendAlertIfNeeded(monitoring.getCurrentAverage(method, path), duration, path,
99                          method);
100             }
101         }
102     }
103 
104     /**
105      * Send an email alert if the time to answer request it too long
106      *
107      * @param average time average
108      * @param currentDuration current duration
109      * @param path resource name
110      * @param method method name
111      */
112     private void sendAlertIfNeeded(final double average, final double currentDuration,
113             final String path, final Method method) {
114         final double elevation = (currentDuration - average * 1000) * 100.0 / average;
115         if (elevation > THRESHOLD_SPEED_PERCENT) {
116             final EmailSettings email = EmailSettings.getInstance();
117             final String subject = "Speed performance alert for " + path;
118             final String msg = "Dear administrator,\nthe speed performance of the " + "application "
119                     + path + "" + " has been reduced of " + String.format("%.02f", elevation)
120                     + "% using the method " + method.getName() + ".\n\n" + "Details:\n"
121                     + "--------\n" + " * Average : " + average + "\n" + " * current : "
122                     + currentDuration + "\n";
123             email.sendMessage(subject, msg);
124         }
125     }
