130     /**
131      * DataCite recommends that only the following characters are used within a
132      * DOI name:
133      * <ul>
134      * <li>0-9</li>
135      * <li>a-z</li>
136      * <li>A-Z</li>
137      * <li>- (dash)</li>
138      * <li>. (dot)</li>
139      * <li>_ (underscore)</li>
140      * <li>+ (plus)</li>
141      * <li>: (colon)</li>
142      * <li>/ (slash)</li>
143      * </ul>
144      *
145      * @param test DOI name to test
146      * @throws IllegalArgumentException An exception is thrown when at least one
147      * character is not part of 0-9a-zA-Z\\-._+:/ of a DOI name
148      */
149     public static void checkIfAllCharsAreValid(final String test) {
150         if (!test.matches("^[0-9a-zA-Z\\-._+:/\\s]+$")) {
151             throw new IllegalArgumentException("Only these characters are allowed "
152                     + "0-9a-zA-Z\\-._+:/ in a DOI name");
153         }
154     }

164     /**
165      * Creates a client to handle DataCite server.
166      *
167      * There is special test prefix 10.5072 available to all datacentres. Please
168      * use it for all your testing DOIs. Your real prefix should not be used for
169      * test DOIs. Note that DOIs with test prefix will behave like any other
170      * DOI, e.g. they can be normally resolved. They will not be exposed by
171      * upcoming services like search and OAI, though. Periodically we purge *
172      * all 10.5072 datasets from the system.
173      *
174      * <p>
175      * It is important to understand that the Handle System (the technical
176      * infrastructure for DOIs) is a distributed network system. The consequence
177      * of this manifests is its inherent latency. For example, DOIs have TTL
178      * (time to live) defaulted to 24 hours, so your changes will be visible to
179      * the resolution infrastructure only when the TTL expires. Also, if you
180      * create a DOI and then immediately try to update its URL, you might get
181      * the error message HANDLE NOT EXISTS. This is because it takes some time
182      * for the system to register a handle for a DOI.
183      *
184      * Each API call can have optional query parametertestMode. If set to "true"
185      * or "1" the request will not change the database nor will the DOI handle
186      * will be registered or updated, e.g. POST /doi?testMode=true and the
187      * testing prefix will be used instead of the provided prefix
188      *
189      * @param context Context using
190      * @throws fr.cnes.doi.exception.ClientMdsException Cannot the Datacite
191      * schema
192      */
193     public ClientMDS(final Context context) throws ClientMdsException {
194         super(context.getDataCiteUrl());
195         this.context = context;
196         this.testMode = this.context.hasTestMode() ? TEST_MODE : null;
197     }

199     /**
200      * Creates a client to handle DataCite with a HTTP Basic authentication.
201      *
202      * All the traffic goes via HTTPS - please remember we do not support bare
203      * HTTP. All the requests to this system require HTTP Basic authentication
204      * header. You will get your username and password from your local DataCite
205      * allocator. Each account have some constraints associated with it:
206      * <ul>
207      * <li>you will be allowed to mint DOIs only with prefix assigned to
208      * you</li>
209      * <li>you will be allowed to mint DOIs only with URLs in host domains
210      * assigned to you</li>
211      * <li>you might not be able to mint unlimited number of DOIs, there is a
212      * quota assigned to you by your allocator (the quota can be extended or
213      * lifted though)</li>
214      * </ul>
215      * Each API call can have optional query parametertestMode. If set to "true"
216      * or "1" the request will not change the database nor will the DOI handle
217      * will be registered or updated, e.g. POST /doi?testMode=true.
218      *
219      * @param context Context using
220      * @param login Login
221      * @param pwd password
222      * @throws fr.cnes.doi.exception.ClientMdsException Cannot the Datacite
223      * schema
224      */

226             final String login,
227             final String pwd) throws ClientMdsException {
228         this(context);
229         this.getLog().debug("Authentication with HTTP_BASIC : {}/{}",

231         this.getClient().setChallengeResponse(ChallengeScheme.HTTP_BASIC, login, pwd);

260     /**
261      * Renames the current DOI prefix by the DOI test prefix.
262      *
263      * @param doiName real DOI name
264      * @return the renamed DOI with the test prefix
265      */
266     private String useTestPrefix(final String doiName) {
267         final String[] split = doiName.split("/");
268         split[0] = TEST_DOI_PREFIX;
269         final String testingPrefix = String.join("/", split);
270         final String message = String.format(
271                 "DOI %s has been renamed as %s for testing", doiName, testingPrefix
272         );
273         this.getLog().warn(message);
274         return testingPrefix;
275     }

301      * Creates the URL to query.
302      *
303      * @param segment segment to add to the end point service
304      * @param doiName doi name
305      * @return the URL to query
306      */
307     private Reference createReferenceWithDOI(final String segment,
308             final String doiName) {
309         final String requestDOI = getDoiAccorgindToContext(doiName);
310         final Reference ref = createReference(segment);
311         final String[] split = requestDOI.split("/");
312         for (final String segmentUri : split) {
313             ref.addSegment(segmentUri);
314         }
315         return ref;
316     }

327         return this.context.hasDoiTestPrefix() ? useTestPrefix(doiName) : doiName;

330     /**
331      * Checks the input parameters and specially the validity of the DOI name.
332      * The real prefix is replaced by the test prefix in DEV, POST_DEV and
333      * PRE_PROD context. The DOI prefix may replace according to the
334      * {@link ClientMDS#context}.
335      *
336      * @param form query form
337      * @throws IllegalArgumentException An exception is thrown when doi and url
338      * are not provided or when one character at least in the DOI name is not
339      * valid
340      */
341     private void checkInputForm(final Form form) throws IllegalArgumentException {
342         final Map<String, String> map = form.getValuesMap();
343         if (map.containsKey(POST_DOI) && map.containsKey(POST_URL)) {
344             final String doiName = form.getFirstValue(POST_DOI);
345             checkIfAllCharsAreValid(doiName);
346             form.set(POST_DOI, getDoiAccorgindToContext(doiName));
347         } else {
348             throw new IllegalArgumentException(MessageFormat.format(
349                     "{0} and {1} parameters are required",
350                     POST_DOI, POST_URL)
351             );
352         }
353     }

373     /**
374      * This request returns an URL associated with a given DOI. A 200 status is
375      * an operation successful. A 204 status means no content (DOI is known to
376      * MDS, but is not minted (or not resolvable e.g. due to handle's latency)
377      * The DOI prefix may replace according to the {@link ClientMDS#context}.
378      *
379      * @param doiName DOI name
380      * @return an URL or no content (DOI is known to MDS, but is not minted (or
381      * not resolvable e.g. due to handle's latency))
382      * @throws ClientMdsException - if an error happens <ul>
383      * <li>401 Unauthorized - no login</li>
384      * <li>403 - login problem or dataset belongs to another party</li>
385      * <li>404 Not Found - DOI does not exist in our database</li>
386      * <li>500 Internal Server Error - server internal error, try later and if
387      * problem persists please contact us</li>
388      * </ul>
389      * @see "https://mds.datacite.org/static/apidoc#tocAnchor-13"
390      */
391     public String getDoi(final String doiName) throws ClientMdsException {
392         final Reference url = createReferenceWithDOI(DOI_RESOURCE, doiName);
393         this.getLog().info("GET {0}", url.toString());
394 
395         this.getClient().setReference(url);
396         Representation rep;
397         try {
398             rep = this.getClient().get();
399             return (rep == null) ? "" : this.getText(rep);
400         } catch (ResourceException ex) {
401             throw new ClientMdsException(ex.getStatus(), ex.getMessage(), this.getClient().
402                     getResponseEntity(), ex);
403         } finally {
404             this.getClient().release();
405         }
406     }

408     /**
409      * Will mint new DOI if specified DOI doesn't exist.
410      *
411      * This method will attempt to update URL if you specify existing DOI.
412      * Standard domains and quota restrictions check will be performed. A
413      * Datacentre's doiQuotaUsed will be increased by 1. A new record in
414      * Datasets will be created when 201 Created status is returned.
415      *
416      * The DOI prefix may replace according to the {@link ClientMDS#context}.
417      *
418      * @param form A form with the following attributes doi and url
419      * @return short explanation of status code e.g. CREATED,
420      * HANDLE_ALREADY_EXISTS etc
421      * @throws ClientMdsException - if an error happens <ul>
422      * <li>400 Bad Request - request body must be exactly two lines: DOI and
423      * URL; wrong domain, wrong prefix</li>
424      * <li>401 Unauthorized - no login</li>
425      * <li>403 Forbidden - login problem, quota exceeded</li>
426      * <li>412 Precondition failed - metadata must be uploaded first</li>
427      * <li>500 Internal Server Error - server internal error, try later and if
428      * problem persists please contact us</li>
429      * </ul>
430      * @see "https://mds.datacite.org/static/apidoc#tocAnchor-15"
431      */
432     public String createDoi(final Form form) throws ClientMdsException {
433         try {
434             this.checkInputForm(form);
435             final Reference url = createReference(DOI_RESOURCE);
436             this.getLog().debug("POST {0}", url.toString());
437             final Representation rep = createRequest(url, form);
438             return getText(rep);
439         } catch (ResourceException ex) {
440             throw new ClientMdsException(ex.getStatus(), ex.getMessage(), this.getClient().
441                     getResponseEntity(), ex);
442         } finally {
443             this.getClient().release();
444         }
445     }

447     /**
448      * Creates the request and requests the DOI creation
449      *
450      * @param url url
451      * @param form form
452      * @return representation of the response
453      */
454     private Representation createRequest(final Reference url, final Form form) {
455         this.getClient().setReference(url);
456         String requestBody = POST_DOI + "=" + form.getFirstValue(POST_DOI) + "\n"
457                 + POST_URL + "=" + form.getFirstValue(POST_URL);
458         requestBody = new String(requestBody.getBytes(
459                 StandardCharsets.UTF_8),
460                 StandardCharsets.UTF_8
461         );
462         final Map<String, Object> requestAttributes = this.getClient().getRequestAttributes();
463         requestAttributes.put("charset", StandardCharsets.UTF_8);
464         requestAttributes.put("Content-Type", "text/plain");
465         this.getLog().info("POST {} with parameters {}", url, requestBody);
466         return this.getClient().post(requestBody, MediaType.TEXT_PLAIN);
467     }

566     /**
567      * Creates metadata with 201 status when operation successful. The DOI
568      * prefix may replace according to the {@link ClientMDS#context}.
569      *
570      * The method is synchronized because marshall method is not thread-safe.
571      *
572      * @param entity Metadata
573      * @return short explanation of status code e.g. CREATED,
574      * HANDLE_ALREADY_EXISTS etc
575      * @throws ClientMdsException - if an error happens <ul>
576      * <li>400 Bad Request - invalid XML, wrong prefix</li>
577      * <li>401 Unauthorized - no login</li>
578      * <li>403 Forbidden - login problem, quota exceeded</li>
579      * <li>500 Internal Server Error - server internal error, try later and if
580      * problem persists please contact us</li>
581      * </ul>
582      * @see "https://mds.datacite.org/static/apidoc#tocAnchor-18"
583      */
584     public String createMetadata(final Resource entity) throws ClientMdsException {
585         try {
586             final Identifier identifier = entity.getIdentifier();
587             identifier.setValue(getDoiAccorgindToContext(identifier.getValue()));
588             final Reference url = createReference(METADATA_RESOURCE);
589             this.getLog().debug("PUT {}", url.toString());
590             final JaxbRepresentation<Resource> result = new JaxbRepresentation<>(entity);
591             result.setCharacterSet(CharacterSet.UTF_8);
592             result.setMediaType(MediaType.APPLICATION_XML);
593             this.getClient().setReference(url);
594             this.getClient().getRequestAttributes().put("charset", "UTF-8");
595             this.getClient().setMethod(null);
596             final Representation response = this.getClient().post(result);
597             return getText(response);
598         } catch (ResourceException ex) {
599             throw new ClientMdsException(ex.getStatus(), ex.getMessage(), this.getClient().
600                     getResponseEntity(), ex);
601         } finally {
602             this.getClient().release();
603         }
604     }

606     /**
607      * Parses the metadata and returns the Resource object from DataCite.
608      *
609      * The method is synchronized because unmarshall method is not thread-safe.
610      *
611      * @param entity metadata
612      * @return the Resource object from DataCite
613      * @throws ValidationException When validation failed
614      */
615     public synchronized Resource parseMetadata(final Representation entity) throws
616             ValidationException {
617 
618         try {
619             final JaxbRepresentation<Resource> resourceEntity = new JaxbRepresentation<>(entity,
620                     Resource.class);
621             final MyValidationEventHandler validationHandler = new MyValidationEventHandler(this.
622                     getClient().getLogger());
623             resourceEntity.setValidationEventHandler(validationHandler);
624             final Resource resource = resourceEntity.getObject();
625             if (validationHandler.isValid()) {
626                 return resource;
627             } else {
628                 throw new ValidationException(validationHandler.getErrorMsg());
629             }

770      * @return Text entity

772     private Representation createEntity(final Form mediaForm) {
773         final Iterator<Parameter> iter = mediaForm.iterator();
774         StringBuilder entity = new StringBuilder();
775         while (iter.hasNext()) {
776             final Parameter param = iter.next();
777             final String mimeType = param.getName();
778             final String url = param.getValue();

780         }
781         return new StringRepresentation(
782                 entity.toString(),
783                 MediaType.TEXT_PLAIN,
784                 Language.ENGLISH,
785                 CharacterSet.UTF_8
