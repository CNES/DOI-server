130     /**
131      * DataCite recommends that only the following characters are used within a DOI name:
132      * <ul>
133      * <li>0-9</li>
134      * <li>a-z</li>
135      * <li>A-Z</li>
136      * <li>- (dash)</li>
137      * <li>. (dot)</li>
138      * <li>_ (underscore)</li>
139      * <li>+ (plus)</li>
140      * <li>: (colon)</li>
141      * <li>/ (slash)</li>
142      * </ul>
143      *
144      * @param test DOI name to test
145      * @throws IllegalArgumentException An exception is thrown when at least one character is not
146      * part of 0-9a-zA-Z\\-._+:/ of a DOI name
147      */
148     public static void checkIfAllCharsAreValid(final String test) {
149         if (!test.matches("^[0-9a-zA-Z\\-._+:/\\s]+$")) {
150             throw new IllegalArgumentException("Only these characters are allowed "
151                     + "0-9a-zA-Z\\-._+:/ in a DOI name");
152         }
153     }

163     /**
164      * Creates a client to handle DataCite server.
165      *
166      * There is special test prefix 10.5072 available to all datacentres. Please use it for all your
167      * testing DOIs. Your real prefix should not be used for test DOIs. Note that DOIs with test
168      * prefix will behave like any other DOI, e.g. they can be normally resolved. They will not be
169      * exposed by upcoming services like search and OAI, though. Periodically we purge * all 10.5072
170      * datasets from the system.
171      *
172      * <p>
173      * It is important to understand that the Handle System (the technical infrastructure for DOIs)
174      * is a distributed network system. The consequence of this manifests is its inherent latency.
175      * For example, DOIs have TTL (time to live) defaulted to 24 hours, so your changes will be
176      * visible to the resolution infrastructure only when the TTL expires. Also, if you create a DOI
177      * and then immediately try to update its URL, you might get the error message HANDLE NOT
178      * EXISTS. This is because it takes some time for the system to register a handle for a DOI.
179      *
180      * Each API call can have optional query parametertestMode. If set to "true" or "1" the request
181      * will not change the database nor will the DOI handle will be registered or updated, e.g. POST
182      * /doi?testMode=true and the testing prefix will be used instead of the provided prefix
183      *
184      * @param context Context using
185      * @throws fr.cnes.doi.exception.ClientMdsException Cannot the Datacite schema
186      */
187     public ClientMDS(final Context context) throws ClientMdsException {
188         super(context.getDataCiteUrl());
189         this.context = context;
190         this.testMode = this.context.hasTestMode() ? TEST_MODE : null;
191     }

193     /**
194      * Creates a client to handle DataCite with a HTTP Basic authentication.
195      *
196      * All the traffic goes via HTTPS - please remember we do not support bare HTTP. All the
197      * requests to this system require HTTP Basic authentication header. You will get your username
198      * and password from your local DataCite allocator. Each account have some constraints
199      * associated with it:
200      * <ul>
201      * <li>you will be allowed to mint DOIs only with prefix assigned to you</li>
202      * <li>you will be allowed to mint DOIs only with URLs in host domains assigned to you</li>
203      * <li>you might not be able to mint unlimited number of DOIs, there is a quota assigned to you
204      * by your allocator (the quota can be extended or lifted though)</li>
205      * </ul>
206      * Each API call can have optional query parametertestMode. If set to "true" or "1" the request
207      * will not change the database nor will the DOI handle will be registered or updated, e.g. POST
208      * /doi?testMode=true.
209      *
210      * @param context Context using
211      * @param login Login
212      * @param pwd password
213      * @throws fr.cnes.doi.exception.ClientMdsException Cannot the Datacite schema
214      */
215     public ClientMDS(final Context context,
216             final String login,
217             final String pwd) throws ClientMdsException {
218         this(context);
219         this.getLog().debug("Authentication with HTTP_BASIC : {}/{}",
220                 login, Utils.transformPasswordToStars(pwd));
221         this.getClient().setChallengeResponse(ChallengeScheme.HTTP_BASIC, login, pwd);
222     }

249     /**
250      * Renames the current DOI prefix by the DOI test prefix.
251      *
252      * @param doiName real DOI name
253      * @return the renamed DOI with the test prefix
254      */
255     private String useTestPrefix(final String doiName) {
256         final String[] split = doiName.split("/");
257         split[0] = TEST_DOI_PREFIX;
258         final String testingPrefix = String.join("/", split);
259         final String message = String.format(
260                 "DOI %s has been renamed as %s for testing", doiName, testingPrefix
261         );
262         this.getLog().warn(message);
263         return testingPrefix;
264     }

291      * @param segment segment to add to the end point service
292      * @param doiName doi name
293      * @return the URL to query
294      */
295     private Reference createReferenceWithDOI(final String segment,
296             final String doiName) {
297         final String requestDOI = getDoiAccorgindToContext(doiName);
298         final Reference ref = createReference(segment);
299         final String[] split = requestDOI.split("/");

302         }
303         return ref;

310      * @param doiName DOI name

313     private String getDoiAccorgindToContext(final String doiName) {
314         return this.context.hasDoiTestPrefix() ? useTestPrefix(doiName) : doiName;

317     /**
318      * Checks the input parameters and specially the validity of the DOI name. The real prefix is
319      * replaced by the test prefix in DEV, POST_DEV and PRE_PROD context. The DOI prefix may replace
320      * according to the {@link ClientMDS#context}.
321      *
322      * @param form query form
323      * @throws IllegalArgumentException An exception is thrown when doi and url are not provided or
324      * when one character at least in the DOI name is not valid
325      */
326     private void checkInputForm(final Form form) throws IllegalArgumentException {
327         final Map<String, String> map = form.getValuesMap();
328         if (map.containsKey(POST_DOI) && map.containsKey(POST_URL)) {
329             final String doiName = form.getFirstValue(POST_DOI);
330             checkIfAllCharsAreValid(doiName);
331             form.set(POST_DOI, getDoiAccorgindToContext(doiName));
332         } else {
333             throw new IllegalArgumentException(MessageFormat.format(
334                     "{0} and {1} parameters are required",
335                     POST_DOI, POST_URL)
336             );
337         }
338     }

358     /**
359      * This request returns an URL associated with a given DOI. A 200 status is an operation
360      * successful. A 204 status means no content (DOI is known to MDS, but is not minted (or not
361      * resolvable e.g. due to handle's latency) The DOI prefix may replace according to the
362      * {@link ClientMDS#context}.
363      *
364      * @param doiName DOI name
365      * @return an URL or no content (DOI is known to MDS, but is not minted (or not resolvable e.g.
366      * due to handle's latency))
367      * @throws ClientMdsException - if an error happens <ul>
368      * <li>401 Unauthorized - no login</li>
369      * <li>403 - login problem or dataset belongs to another party</li>
370      * <li>404 Not Found - DOI does not exist in our database</li>
371      * <li>500 Internal Server Error - server internal error, try later and if problem persists
372      * please contact us</li>
373      * </ul>
374      * @see "https://mds.datacite.org/static/apidoc#tocAnchor-13"
375      */
376     public String getDoi(final String doiName) throws ClientMdsException {
377         final Reference url = createReferenceWithDOI(DOI_RESOURCE, doiName);
378         this.getLog().info("GET {0}", url.toString());
379 
380         this.getClient().setReference(url);
381         Representation rep;
382         try {
383             rep = this.getClient().get();
384             return (rep == null) ? "" : this.getText(rep);
385         } catch (ResourceException ex) {
386             throw new ClientMdsException(ex.getStatus(), ex.getMessage(), this.getClient().
387                     getResponseEntity(), ex);
388         } finally {
389             this.getClient().release();
390         }
391     }

393     /**
394      * Will mint new DOI if specified DOI doesn't exist.
395      *
396      * This method will attempt to update URL if you specify existing DOI. Standard domains and
397      * quota restrictions check will be performed. A Datacentre's doiQuotaUsed will be increased by
398      * 1. A new record in Datasets will be created when 201 Created status is returned.
399      *
400      * The DOI prefix may replace according to the {@link ClientMDS#context}.
401      *
402      * @param form A form with the following attributes doi and url
403      * @return short explanation of status code e.g. CREATED, HANDLE_ALREADY_EXISTS etc
404      * @throws ClientMdsException - if an error happens <ul>
405      * <li>400 Bad Request - request body must be exactly two lines: DOI and URL; wrong domain,
406      * wrong prefix</li>
407      * <li>401 Unauthorized - no login</li>
408      * <li>403 Forbidden - login problem, quota exceeded</li>
409      * <li>412 Precondition failed - metadata must be uploaded first</li>
410      * <li>500 Internal Server Error - server internal error, try later and if problem persists
411      * please contact us</li>
412      * </ul>
413      * @see "https://mds.datacite.org/static/apidoc#tocAnchor-15"
414      */
415     public String createDoi(final Form form) throws ClientMdsException {
416         try {
417             this.checkInputForm(form);
418             final Reference url = createReference(DOI_RESOURCE);
419             this.getLog().debug("POST {0}", url.toString());
420             final Representation rep = createRequest(url, form);
421             return getText(rep);
422         } catch (ResourceException ex) {
423             throw new ClientMdsException(ex.getStatus(), ex.getMessage(), this.getClient().
424                     getResponseEntity(), ex);
425         } finally {
426             this.getClient().release();
427         }
428     }

430     /**
431      * Creates the request and requests the DOI creation
432      *
433      * @param url url
434      * @param form form
435      * @return representation of the response
436      */
437     private Representation createRequest(final Reference url, final Form form) {
438         this.getClient().setReference(url);
439         String requestBody = POST_DOI + "=" + form.getFirstValue(POST_DOI) + "\n"
440                 + POST_URL + "=" + form.getFirstValue(POST_URL);
441         requestBody = new String(requestBody.getBytes(
442                 StandardCharsets.UTF_8),
443                 StandardCharsets.UTF_8
444         );
445         final Map<String, Object> requestAttributes = this.getClient().getRequestAttributes();
446         requestAttributes.put("charset", StandardCharsets.UTF_8);
447         requestAttributes.put("Content-Type", "text/plain");
448         this.getLog().info("POST {} with parameters {}", url, requestBody);
449         return this.getClient().post(requestBody, MediaType.TEXT_PLAIN);
450     }

541     /**
542      * Creates metadata with 201 status when operation successful. The DOI prefix may replace
543      * according to the {@link ClientMDS#context}.
544      *
545      * The method is synchronized because marshall method is not thread-safe.
546      *
547      * @param entity Metadata
548      * @return short explanation of status code e.g. CREATED, HANDLE_ALREADY_EXISTS etc
549      * @throws ClientMdsException - if an error happens <ul>
550      * <li>400 Bad Request - invalid XML, wrong prefix</li>
551      * <li>401 Unauthorized - no login</li>
552      * <li>403 Forbidden - login problem, quota exceeded</li>
553      * <li>500 Internal Server Error - server internal error, try later and if problem persists
554      * please contact us</li>
555      * </ul>
556      * @see "https://mds.datacite.org/static/apidoc#tocAnchor-18"
557      */
558     public String createMetadata(final Resource entity) throws ClientMdsException {
559         try {
560             final Identifier identifier = entity.getIdentifier();
561             identifier.setValue(getDoiAccorgindToContext(identifier.getValue()));
562             final Reference url = createReference(METADATA_RESOURCE);
563             this.getLog().debug("PUT {}", url.toString());
564             final JaxbRepresentation<Resource> result = new JaxbRepresentation<>(entity);
565             result.setCharacterSet(CharacterSet.UTF_8);
566             result.setMediaType(MediaType.APPLICATION_XML);
567             this.getClient().setReference(url);
568             this.getClient().getRequestAttributes().put("charset", "UTF-8");
569             this.getClient().setMethod(null);
570             final Representation response = this.getClient().post(result);
571             return getText(response);
572         } catch (ResourceException ex) {
573             throw new ClientMdsException(ex.getStatus(), ex.getMessage(), this.getClient().
574                     getResponseEntity(), ex);
575         } finally {
576             this.getClient().release();
577         }
578     }

580     /**
581      * Parses the metadata and returns the Resource object from DataCite.
582      *
583      * The method is synchronized because unmarshall method is not thread-safe.
584      *
585      * @param entity metadata
586      * @return the Resource object from DataCite
587      * @throws ValidationException When validation failed
588      */
589     public synchronized Resource parseMetadata(final Representation entity) throws
590             ValidationException {
591 
592         try {
593             final JaxbRepresentation<Resource> resourceEntity = new JaxbRepresentation<>(entity,
594                     Resource.class);
595             final MyValidationEventHandler validationHandler = new MyValidationEventHandler(this.
596                     getClient().getLogger());
597             resourceEntity.setValidationEventHandler(validationHandler);
598             final Resource resource = resourceEntity.getObject();
599             if (validationHandler.isValid()) {
600                 return resource;
601             } else {
602                 throw new ValidationException(validationHandler.getErrorMsg());
603             }
604         } catch (IOException | JAXBException ex) {
605             throw new ValidationException("Cannot read the metadata", ex);
606         }
607     }

742         StringBuilder entity = new StringBuilder();
