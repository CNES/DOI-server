327     /**
328      * Returns the list of DOI projects
329      *
330      * @param statement Query to T_DOI_PROJECT
331      * @return the DOI projects related to the query
332      * @throws SQLException When an SQL execption occurs
333      */
334     private List<DOIProject> getDOIProjects(final PreparedStatement statement) throws SQLException {
335         LOGGER.traceEntry("Parameter\n\t statement: {}", statement.toString());
336         LOGGER.debug(statement.toString());
337         final List<DOIProject> projects = new ArrayList<>();
338         try (final ResultSet rs = statement.executeQuery()) {
339             while (rs.next()) {
340                 final DOIProject doiproject = new DOIProject();
341                 doiproject.setSuffix(rs.getInt(FIELD_PROJECT_SUFFIX));
342                 doiproject.setProjectname(rs.getString(FIELD_PROJECTNAME));
343                 projects.add(doiproject);
344             }
345         }
346         LOGGER.debug("{} DOI projects found", projects.size());
347         return LOGGER.traceExit(projects);
348     }

351      * Returns the list of DOI users
352      *
353      * @param statement Query to T_DOI_USERS
354      * @return the DOI users related to the query
355      * @throws SQLException When an SQL execption occurs
356      */
357     private List<DOIUser> getDOIUSers(final PreparedStatement statement) throws SQLException {
358         LOGGER.traceEntry("Parameter\n\t statement: {}", statement.toString());
359         LOGGER.debug(statement.toString());
360         final List<DOIUser> users = new ArrayList<>();
361         try (final ResultSet rs = statement.executeQuery()) {
362             while (rs.next()) {
363                 final DOIUser doiuser = new DOIUser();
364                 doiuser.setUsername(rs.getString(FIELD_USERNAME));
365                 doiuser.setAdmin(rs.getBoolean(FIELD_ADMIN));
366                 doiuser.setEmail(rs.getString(FIELD_EMAIL));
367                 users.add(doiuser);
368             }
369         }
370         LOGGER.debug("{} DOI users found", users.size());
371         return LOGGER.traceExit(users);
372     }

444     /**
445      * Close the statements and free the connection.
446      *
447      * @param conn connection to closeAndRelease
448      * @param statements statements to closeAndRelease
449      */
450     private void closeAndRelease(final Connection conn, final PreparedStatement... statements) {
451         for (final PreparedStatement statement : statements) {
452             if (statement != null) {
453                 try {
454                     statement.close();
455                 } catch (SQLException e) {
456                     LOGGER.error("Unable to close statement", e);
457                 }
458             }
459         }
460         if (conn != null) {
461             try {
462                 conn.close();
463             } catch (SQLException e) {
464                 LOGGER.error("Unable to close connection to database", e);
465             }
466         }
467     }

634      * {@inheritDoc}
635      */
636     @Override
637     @SuppressFBWarnings(value = "OBL_UNSATISFIED_OBLIGATION",
638             justification = "Cleans up with closeAndRelease method")
639     public void addDOIProjectToUser(final String username, final int suffix) throws DOIDbException {
640         LOGGER.traceEntry("Parameters:\n\tusername:{}\n\tsuffix:{}", username, suffix);
641         Connection conn = null;
642         PreparedStatement userStatement = null;
643         PreparedStatement projectStatement = null;
644         PreparedStatement assignationStatement = null;
645         try {
646             conn = dbConnector.getConnection();
647 
648             userStatement = conn.prepareStatement(SELECT_EXISTS_USERNAME);
649             userStatement.setString(1, username);
650             final boolean isUserExist = isQueryExist(userStatement);
651 
652             projectStatement = conn.prepareStatement(SELECT_EXISTS_SUFFIX);
653             projectStatement.setInt(1, suffix);
654             final boolean isProjectExist = isQueryExist(projectStatement);
655 
656             if (!isUserExist || !isProjectExist) {
657                 throw LOGGER.throwing(
658                         Level.FATAL,
659                         new DOIDbException(
660                                 "An exception occured when calling addDOIProjectToUser: user " + username
661                                 + " or project " + suffix + " don't exist in doi database", null)
662                 );
663             }
664             assignationStatement = conn.prepareStatement(INSERT_DOI_ASSIGN);
665             assignationStatement.setString(1, username);
666             assignationStatement.setInt(2, suffix);
667             this.updateQueries(assignationStatement);
668         } catch (SQLException e) {
