323     /**
324      * Returns the list of DOI projects
325      *
326      * @param statement Query to T_DOI_PROJECT
327      * @return the DOI projects related to the query
328      * @throws SQLException When an SQL execption occurs
329      */
330     private List<DOIProject> getDOIProjects(final PreparedStatement statement) throws SQLException {
331         LOGGER.traceEntry("Parameter\n\t statement: {}", statement.toString());
332         LOGGER.debug(statement.toString());
333         final List<DOIProject> projects = new ArrayList<>();
334         try (final ResultSet rs = statement.executeQuery()) {
335             while (rs.next()) {
336                 final DOIProject doiproject = new DOIProject();
337                 doiproject.setSuffix(rs.getInt(FIELD_PROJECT_SUFFIX));
338                 doiproject.setProjectname(rs.getString(FIELD_PROJECTNAME));
339                 projects.add(doiproject);
340             }
341         }
342         LOGGER.debug("{} DOI projects found", projects.size());
343         return LOGGER.traceExit(projects);
344     }

347      * Returns the list of DOI users
348      *
349      * @param statement Query to T_DOI_USERS
350      * @return the DOI users related to the query
351      * @throws SQLException When an SQL execption occurs
352      */
353     private List<DOIUser> getDOIUSers(final PreparedStatement statement) throws SQLException {
354         LOGGER.traceEntry("Parameter\n\t statement: {}", statement.toString());
355         LOGGER.debug(statement.toString());
356         final List<DOIUser> users = new ArrayList<>();
357         try (final ResultSet rs = statement.executeQuery()) {
358             while (rs.next()) {
359                 final DOIUser doiuser = new DOIUser();
360                 doiuser.setUsername(rs.getString(FIELD_USERNAME));
361                 doiuser.setAdmin(rs.getBoolean(FIELD_ADMIN));
362                 doiuser.setEmail(rs.getString(FIELD_EMAIL));
363                 users.add(doiuser);
364             }
365         }
366         LOGGER.debug("{} DOI users found", users.size());
367         return LOGGER.traceExit(users);
368     }

440     /**
441      * Close the statements and free the connection.
442      *
443      * @param conn connection to closeAndRelease
444      * @param statements statements to closeAndRelease
445      */
446     private void closeAndRelease(final Connection conn, final PreparedStatement... statements) {
447         for (final PreparedStatement statement : statements) {
448             if (statement != null) {
449                 try {
450                     statement.close();
451                 } catch (SQLException e) {
452                     LOGGER.error("Unable to close statement", e);
453                 }
454             }
455         }
456         if (conn != null) {
457             try {
458                 conn.close();
459             } catch (SQLException e) {
460                 LOGGER.error("Unable to close connection to database", e);
461             }
462         }
463     }

631      */
632     @Override
633     @SuppressFBWarnings(value = "OBL_UNSATISFIED_OBLIGATION",
634             justification = "Cleans up with closeAndRelease method")
635     public void addDOIProjectToUser(final String username, final int suffix) throws DOIDbException {
636         LOGGER.traceEntry("Parameters:\n\tusername:{}\n\tsuffix:{}", username, suffix);
637         Connection conn = null;
638         PreparedStatement userStatement = null;
639         PreparedStatement projectStatement = null;
640         PreparedStatement assignationStatement = null;
641         try {
642             conn = dbConnector.getConnection();
643 
644             userStatement = conn.prepareStatement(SELECT_EXISTS_USERNAME);
645             userStatement.setString(1, username);
646             final boolean isUserExist = isQueryExist(userStatement);
647 
648             projectStatement = conn.prepareStatement(SELECT_EXISTS_SUFFIX);
649             projectStatement.setInt(1, suffix);
650             final boolean isProjectExist = isQueryExist(projectStatement);
651 
652             if (!isUserExist || !isProjectExist) {
653                 throw LOGGER.throwing(
654                         Level.FATAL,
655                         new DOIDbException(
656                                 "An exception occured when calling addDOIProjectToUser: user " + username
657                                 + " or project " + suffix + " don't exist in doi database", null)
658                 );
659             }
660             assignationStatement = conn.prepareStatement(INSERT_DOI_ASSIGN);
661             assignationStatement.setString(1, username);
662             assignationStatement.setInt(2, suffix);
663             this.updateQueries(assignationStatement);
664         } catch (SQLException e) {
665             throw LOGGER.throwing(
666                     Level.ERROR,
667                     new DOIDbException("An exception occured when calling addDOIProjectToUser", e)
