115     @Override
116     public void setConfiguration(final Object configuration) {
117         this.conf = (Map<String, String>) configuration;
118         final String dbUrl = this.conf.get(DB_URL);
119         final String dbUser = this.conf.get(DB_USER);
120         final String dbPwd = this.conf.get(DB_PWD);
121         if (this.conf.containsKey(DB_MIN_IDLE_CONNECTIONS)) {
122             this.options.put(DB_MIN_IDLE_CONNECTIONS,
123                     Integer.valueOf(this.conf.get(DB_MIN_IDLE_CONNECTIONS)));
124         }
125         if (this.conf.containsKey(DB_MAX_IDLE_CONNECTIONS)) {
126             this.options.put(DB_MAX_IDLE_CONNECTIONS,
127                     Integer.valueOf(this.conf.get(DB_MAX_IDLE_CONNECTIONS)));
128         }
129         if (this.conf.containsKey(DB_MAX_ACTIVE_CONNECTIONS)) {
130             this.options.put(DB_MAX_ACTIVE_CONNECTIONS,
131                     Integer.valueOf(this.conf.get(DB_MAX_ACTIVE_CONNECTIONS)));
132         }
133         LOG.info("[CONF] Plugin database URL : {}", dbUrl);
134         LOG.info("[CONF] Plugin database user : {}", dbUser);
135         LOG.info("[CONF] Plugin database password : {}", Utils.transformPasswordToStars(dbPwd));
136         LOG.info("[CONF] Plugin options : {}", options);
137 
138         this.configured = true;
139     }

171     /**
172      * {@inheritDoc }
173      */
174     @Override
175     public boolean addUserToRole(final String user, final int role) {
176         boolean isAdded = false;
177         try {
178             das.addDOIProjectToUser(user, role);
179             isAdded = true;
180             LOG.info("The user {} is added to role {} for {}.", user, role,
181                     Application.getCurrent().getName());
182 
183             final User userFromRealm = REALM.findUser(user);
184             final Role roleFromRealm = new Role(Application.getCurrent(), String.valueOf(role),
185                     "Role " + String.valueOf(role) + " for " + Application.getCurrent().getName());
186             REALM.map(userFromRealm, roleFromRealm);
187 
188         } catch (DOIDbException e) {
189             LOG.fatal("An error occured while trying to add user " + user + " to project " + role,
190                     e);
191         }
192         return isAdded;
193     }

195     /**
196      * {@inheritDoc }
197      */
198     @Override
199     public boolean removeUserToRole(final String user, final int role) {
200         boolean isRemoved = false;
201         try {
202             das.removeDOIProjectFromUser(user, role);
203             isRemoved = true;
204             LOG.info("The user {} is removed from role {} for {}.", user, role, Application.
205                     getCurrent().getName());
206 
207             final User userFromRealm = REALM.findUser(user);
208             final Set<Role> rolesFromRealm = REALM.findRoles(userFromRealm);
209             for (final Role r : rolesFromRealm) {
210                 if (r.getName().equals(String.valueOf(role))) {
211                     REALM.unmap(userFromRealm, r);
212                 }
213             }
214 
215         } catch (DOIDbException e) {
216             LOG.fatal("An error occured while trying to remove user " + user + " from project "
217                     + role, e);
218         }
219         return isRemoved;
220     }

222     /**
223      * Sets the user to the admin group.
224      *
225      * @param user username
226      * @throws DOIDbException - if a Database error occurs
227      */
228     private void setUserToAdminGroupInDB(final String user) throws DOIDbException {
229         das.setAdmin(user);
230         LOG.info("The user {} is added to admin group.", user);
231         EmailSettings.getInstance().sendMessage("[DOI] Admin group",
232                 "The user " + user + " has been added to the administror group.");
233         final User userFromRealm = REALM.findUser(user);
234         if (!REALM.getRootGroups().get(0).getMemberUsers().contains(userFromRealm)) {
235             REALM.getRootGroups().get(0).getMemberUsers().add(userFromRealm);
236         }
237     }

428     /**
429      * {@inheritDoc}
430      */
431     @Override
432     public void release() {
433         final List<User> users = REALM.getUsers();
434         // we unmap all roles for each user
435         for (final User user : users) {
436             final Set<Role> roles = REALM.findRoles(user);
437             final Iterator<Role> roleIter = roles.iterator();
438             while (roleIter.hasNext()) {
439                 final Role role = roleIter.next();
440                 REALM.unmap(user, role);
441             }
442         }
443         try {
444             if (this.das != null) {
445                 this.das.close();
446             }
447         } catch (DOIDbException ex) {
448         }
449         this.configured = false;
450     }
