33  @Requirement(reqId = Requirement.DOI_ARCHI_030, reqName = Requirement.DOI_ARCHI_030_NAME)

36      /**
37       * Map the name of an interface to the name of a corresponding concrete implementation class.
38       */
39      private static final Map<String, String> PLUGINS_IMPL = new ConcurrentHashMap<>();

41      /**
42       * Settings.
43       */
44      private static final Map<String, String> SETTINGS = new ConcurrentHashMap<>();

46      /**
47       * Stores the instances of plugin.

49      private static final Map<String, Object> CONFIG = new ConcurrentHashMap<>();

51      /**
52       * Loads the path of plugins from Settings.
53       *
54       * @param settings config settings
55       */
56      public static void init(final Map<String, String> settings) {
57          final String userRealPlugin = settings.getOrDefault(Consts.PLUGIN_USER_GROUP_MGT, "");
58          final String projectSuffixPlugin = settings.getOrDefault(Consts.PLUGIN_PROJECT_SUFFIX, "");
59          final String tokenPlugin = settings.getOrDefault(Consts.PLUGIN_TOKEN, "");
60          final String authPlugin = settings.getOrDefault(Consts.PLUGIN_AUTHENTICATION, "");
61          PLUGINS_IMPL.put(Consts.PLUGIN_USER_GROUP_MGT, userRealPlugin);
62          PLUGINS_IMPL.put(Consts.PLUGIN_PROJECT_SUFFIX, projectSuffixPlugin);
63          PLUGINS_IMPL.put(Consts.PLUGIN_TOKEN, tokenPlugin);
64          PLUGINS_IMPL.put(Consts.PLUGIN_AUTHENTICATION, authPlugin);
65          SETTINGS.putAll(settings);
66      }

68      /**
69       * Returns the concrete implementation of the user management interface.
70       *
71       * @return the plugin
72       */
73      public static AbstractUserRolePluginHelper getUserManagement() {
74          final String implClassName = PLUGINS_IMPL.get(Consts.PLUGIN_USER_GROUP_MGT);
75          final AbstractUserRolePluginHelper plugin = getPlugin(implClassName);
76          if (!plugin.isConfigured()) {
77              plugin.setConfiguration(SETTINGS);
78              plugin.initConnection();
79          }
80          return plugin;
81      }

128     /**
129      * Checks if the key is a password in the class keywordClassName related to the configuration
130      * file
131      *
132      * @param keywordClassName plugin related to the configuration file
133      * @param key keyword
134      * @return True when key is a password otherwise false
135      * @throws DoiRuntimeException When an error occurs
136      */
137     public static boolean isPassword(final String keywordClassName, final String key) throws
138             DoiRuntimeException {
139         try {
140             final String implClassName = PLUGINS_IMPL.get(keywordClassName);
141             final Class implClass = Class.forName(implClassName);
142             final Method method = implClass.getMethod("isPassword", String.class);
143             final Object obj = method.invoke(null, key);
144             return Boolean.getBoolean(String.valueOf(obj));
145         } catch (ClassNotFoundException | NoSuchMethodException | SecurityException
146                 | IllegalAccessException | IllegalArgumentException
147                 | InvocationTargetException ex) {
148             throw new DoiRuntimeException(ex);
149         }
150     }
151 
152     /**
153      * Returns the plugin according to its implementation class.
154      *
155      * @param <T> the plugin
156      * @param implClassName implementation class
157      * @return the plugin
158      */
159     private static <T> T getPlugin(final String implClassName) {
160         final T plugin;
161         if (CONFIG.get(implClassName) == null) {
162             try {
163                 final Class impClass = Class.forName(implClassName);
164                 plugin = (T) impClass.cast(buildObject(implClassName));
165                 CONFIG.put(implClassName, plugin);
166             } catch (ClassNotFoundException ex) {
167                 throw new DoiRuntimeException(ex);
168             }
169         } else {
170             plugin = (T) CONFIG.get(implClassName);
171         }
172         return plugin;
173     }
174 
175     /**
176      * instantiates the plugin.
177      *
178      * @param aClassName the plugin name
179      * @return instance of aClassName
180      * @throws DoiRuntimeException - if aClassName cannot be instantiated
181      */
182     private static Object buildObject(final String aClassName) throws DoiRuntimeException {
183         final Object result;
184         try {
185             final Class implClass = Class.forName(aClassName);
186             result = implClass.newInstance();
187         } catch (ClassNotFoundException | InstantiationException | IllegalAccessException ex) {
188             throw new DoiRuntimeException(ex);
189         }
190         return result;
191     }
