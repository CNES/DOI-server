39  /**
40   * Provides a unique identifier to the project. This identifier is used as part
41   * of the DOI name.
42   *
43   * @author Jean-Christophe Malapert (jean-christophe.malapert@cnes.fr)
44   */
45  public class SuffixProjectsDoisResource extends AbstractResource {
46  
47      /**
48       * Logger.
49       */
50      private volatile Logger LOG;
51  
52      /**
53       * Suffix of the project.
54       */
55      private volatile String suffixProject;
56  
57      /**
58       * Set-up method that can be overridden in order to initialize the state of
59       * the resource.
60       *
61       * Init by getting the DOI name in the template URL.
62       *
63       * @throws ResourceException - if a problem happens
64       */
65      @Override
66      protected void doInit() throws ResourceException {
67          super.doInit();
68          final AdminApplication app = (AdminApplication) getApplication();
69          LOG = app.getLog();
70          LOG.traceEntry();
71          setDescription("This resource handles the project suffix in the DOI name");
72          this.suffixProject = getResourcePath().replace(
73                  AdminApplication.ADMIN_URI + AdminApplication.SUFFIX_PROJECT_URI + "/", "");
74          final int startIndex = this.suffixProject.indexOf(AdminApplication.DOIS_URI);
75          this.suffixProject = startIndex == -1 ? null : this.suffixProject.substring(0, startIndex);
76          LOG.debug(this.suffixProject);
77  
78          LOG.traceExit();
79      }
80  
81      //TODO requirement
82      /**
83       * Returns the list of dois from the project suffix sent in url.
84       *
85       * @return the list of dois
86       */
87      @Requirement(reqId = Requirement.DOI_SRV_140, reqName = Requirement.DOI_SRV_140_NAME)
88      @Get
89      public List<String> getProjectsNameAsJson() {
90          LOG.traceEntry();
91  //        checkInput(this.suffixProject);
92  
93          final ClientSearchDataCite client;
94          List<String> response = new ArrayList<>();
95  
96          try {
97              client = new ClientSearchDataCite();
98              response = client.getDois(this.suffixProject);
99  
100         } catch (Exception ex) {
101             LOG.error("Error in SuffixProjectsDoisResource while searching for dois in project {}",
102                     this.suffixProject, ex);
103         }
104         return LOG.traceExit(response);
105     }

107     /**
108      * projects representation
109      *
110      * @return Wadl representation for projects
111      */
112     @Requirement(reqId = Requirement.DOI_DOC_010, reqName = Requirement.DOI_DOC_010_NAME)
113     private RepresentationInfo projectsRepresentation() {
114         final RepresentationInfo repInfo = new RepresentationInfo();
115         repInfo.setMediaType(MediaType.APPLICATION_XML);
116         final DocumentationInfo docInfo = new DocumentationInfo();
117         docInfo.setTitle("Projects Representation");
118         docInfo.setTextContent("The representation contains informations about all projects.");
119         repInfo.setDocumentation(docInfo);
120         return repInfo;
121     }
