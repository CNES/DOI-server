43  /**
44   * Provide a resource to delete and rename a project from database.
45   */
46  public class ManageProjectsResource extends AbstractResource {
47  
48      /**
49       * Parameter for the project name {@value #PROJECT_NAME_PARAMETER}. This parameter is send to
50       * create a new identifier for the project.
51       */
52      public static final String PROJECT_NAME_PARAMETER = "newProjectName";
53  
54      /**
55       * Logger.
56       */
57      private volatile Logger LOG;
58  
59      /**
60       * Suffix of the project.
61       */
62      private volatile String suffixProject;
63  
64      /**
65       * Set-up method that can be overridden in order to initialize the state of the resource.
66       *
67       * @throws ResourceException - if a problem happens
68       */
69      @Override
70      protected void doInit() throws ResourceException {
71          super.doInit();
72          final AdminApplication app = (AdminApplication) getApplication();
73          LOG = app.getLog();
74          LOG.traceEntry();
75          setDescription("This resource handles deletion and renaming of a project");
76          this.suffixProject = getAttribute(SUFFIX_PROJECT_NAME_TEMPLATE);
77          LOG.debug(this.suffixProject);
78          LOG.traceExit();
79      }

81      // TODO requirement
82      /**
83       * Rename the project from the project id sent in url.
84       *
85       * @param mediaForm form
86       */
87      @Requirement(reqId = Requirement.DOI_SRV_140, reqName = Requirement.DOI_SRV_140_NAME)
88      @Post
89      public void renameProject(final Form mediaForm) {
90          LOG.traceEntry("Parameters\n\tmediaForm : {}", mediaForm);
91          checkInputs(mediaForm);
92          final String newProjectName = mediaForm.getFirstValue(PROJECT_NAME_PARAMETER);
93          final AbstractProjectSuffixDBHelper manageProjects = PluginFactory.getProjectSuffix();
94          final boolean isRenamed = manageProjects.renameProject(Integer.parseInt(suffixProject),
95                  newProjectName);
96          if (isRenamed) {
97              setStatus(Status.SUCCESS_NO_CONTENT);
98          } else {
99              throw LOG.throwing(new ResourceException(
100                     Status.CLIENT_ERROR_BAD_REQUEST, "Cannot rename project to " + newProjectName));
101         }
102         LOG.traceExit();
103     }

106     /**
107      * Delete the project from database.
108      */
109     @Requirement(reqId = Requirement.DOI_SRV_140, reqName = Requirement.DOI_SRV_140_NAME)
110     @Delete
111     public void deleteProject() {
112         LOG.traceEntry();
113 
114         final ClientSearchDataCite client;
115         final List<String> response = new ArrayList<>();
116         try {
117             client = new ClientSearchDataCite();
118             response.addAll(client.getDois(suffixProject));
119 
120         } catch (Exception ex) {
121             LOG.error(ex + "\n"
122                     + "Error in SuffixProjectsDoisResource while searching for dois in project "
123                     + suffixProject);
124         }
125 
126         final AbstractProjectSuffixDBHelper manageProjects = PluginFactory.getProjectSuffix();
127 
128         // No DOIs have to be attached to a project before deleting it
129         if (!response.isEmpty()) {
130             throw LOG.throwing(new ResourceException(
131                     Status.CLIENT_ERROR_NOT_FOUND, suffixProject + " not found"));
132         } else if (manageProjects.deleteProject(Integer.parseInt(suffixProject))) {
133             setStatus(Status.SUCCESS_NO_CONTENT);
134         } else {
135             throw LOG.throwing(new ResourceException(
136                     Status.SERVER_ERROR_INTERNAL, "Cannot delete the project " + suffixProject));
137         }
138         LOG.traceExit();
139     }

141     /**
142      * Tests if the {@link #PROJECT_NAME_PARAMETER} is set.
143      *
144      * @param mediaForm the parameters
145      * @throws ResourceException - if PROJECT_NAME_PARAMETER is not set
146      */
147     private void checkInputs(final Form mediaForm) throws ResourceException {
148         LOG.traceEntry("Parameters\n\tmediaForm : {}", mediaForm);
149         if (isValueNotExist(mediaForm, PROJECT_NAME_PARAMETER)) {
150             throw LOG.throwing(Level.ERROR, new ResourceException(Status.CLIENT_ERROR_BAD_REQUEST,
151                     PROJECT_NAME_PARAMETER + " parameter must be set"));
152         }
153         LOG.debug("The form is valid");
154         LOG.traceExit();
155     }
