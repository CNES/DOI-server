38  /**
39   * Provide a resource to ask if a user belong to the super user group and another one to remove a
40   * user from the super user group.
41   */
42  public class ManageSuperUserResource extends AbstractResource {
43  
44      /**
45       * Logger.
46       */
47      private volatile Logger LOG;
48  
49      /**
50       * User name.
51       */
52      private volatile String userName;
53  
54      /**
55       * Set-up method that can be overridden in order to initialize the state of the resource.
56       *
57       * @throws ResourceException - if a problem happens
58       */
59      @Override
60      protected void doInit() throws ResourceException {
61          super.doInit();
62          final AdminApplication app = (AdminApplication) getApplication();
63          LOG = app.getLog();
64          LOG.traceEntry();
65          this.userName = getAttribute(USERS_NAME_TEMPLATE);
66          LOG.debug(this.userName);
67          setDescription("This resource handles super user");
68          LOG.traceExit();
69      }
70  
71      // TODO requirement
72      /**
73       * Returns null is user doesn't exist otherwise return true or false if user is admin or not.
74       *
75       * @return boolean.
76       */
77      @Get
78      public boolean isUserExistAndAdmin() {
79          LOG.traceEntry();        
80          final AbstractUserRoleDBHelper manageUsers = PluginFactory.getUserManagement();
81          if (!manageUsers.isUserExist(userName)) {
82              throw LOG.throwing(Level.ERROR, new ResourceException(Status.CLIENT_ERROR_BAD_REQUEST,
83                      "The user " + userName + " does not exist"));
84          }
85          return LOG.traceExit(isInRole(RoleAuthorizer.ROLE_ADMIN));
86      }
87  
88      // TODO requirement
89      /**
90       * Delete the SUPERUSER from database.
91       *
92       */
93      // @Requirement(reqId = Requirement.DOI_SRV_140, reqName =
94      // Requirement.DOI_SRV_140_NAME)
95      @Delete
96      public void deleteSuperUser() {
97          LOG.traceEntry();
98          final AbstractUserRoleDBHelper manageUsers = PluginFactory.getUserManagement();
99          if (manageUsers.unsetUserFromAdminGroup(userName)) {
100             setStatus(Status.SUCCESS_NO_CONTENT);
101         } else {
102             throw LOG.throwing(new ResourceException(
103                     Status.CLIENT_ERROR_BAD_REQUEST, "Can't delete super user " + userName));
104         }
105         LOG.traceExit();
106     }
107 
108     @Override
109     protected void describeGet(final MethodInfo info) {
110         info.setName(Method.GET);
111         info.setDocumentation("Checks if the user is admin");
112         addRequestDocToMethod(info, createQueryParamDoc(
113                 USERS_NAME_TEMPLATE, ParameterStyle.TEMPLATE,
114                 "user name", true, "xs:string")
115         );
116         addResponseDocToMethod(info, createResponseDoc(
117                 Status.SUCCESS_NO_CONTENT, "Operation successful with true or false",
118                 stringRepresentation())
119         );
120 
121         addResponseDocToMethod(info, createResponseDoc(
122                 Status.CLIENT_ERROR_BAD_REQUEST, "The user does not exist",
123                 htmlRepresentation())
124         );
125     }
126 
127     @Override
128     protected void describeDelete(final MethodInfo info) {
129         info.setName(Method.DELETE);
130         info.setDocumentation("Delete a project");
131         addRequestDocToMethod(info, createQueryParamDoc(
132                 USERS_NAME_TEMPLATE, ParameterStyle.TEMPLATE,
133                 "user name", true, "xs:string")
134         );
135         addResponseDocToMethod(info, createResponseDoc(
136                 Status.SUCCESS_NO_CONTENT, "Operation successful",
137                 stringRepresentation())
138         );
139 
140         addResponseDocToMethod(info, createResponseDoc(
141                 Status.CLIENT_ERROR_BAD_REQUEST, "Cannot delete super user",
142                 htmlRepresentation())
143         );
144     }
