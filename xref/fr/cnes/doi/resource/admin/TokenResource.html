118     /**
119      * Creates and stores a token.
120      *
121      * The token creation is based on several actions :
122      * <ul>
123      * <li>creates the {@link fr.cnes.doi.security.TokenSecurity#generate}</li>
124      * <li>stores the token in {@link fr.cnes.doi.db.AbstractTokenDBHelper token database}</li>
125      * </ul>
126      *
127      * @param infoForm submitted information when requesting the token creation
128      * @return the token
129      */
130     @Requirement(reqId = Requirement.DOI_SRV_150, reqName = Requirement.DOI_SRV_150_NAME)
131     @Post
132     public String createToken(final Form infoForm) {
133         LOG.traceEntry("Paramater : {}", infoForm);
134         final Form info = (infoForm == null) ? new Form() : infoForm;
135         try {
136             final String user = this.getClientInfo().getUser().getIdentifier();
137             LOG.debug("Identified user : {}", user);
138             final String userID;
139             if(isInRole(RoleAuthorizer.ROLE_ADMIN)) {
140                 // The admin can generate for everybody
141                 LOG.debug("User {} is admin", user);
142                 userID = info.getFirstValue(IDENTIFIER_PARAMETER, user);
143             } else {
144                 // The token is generated for the identified user.
145                 userID = user;
146             }
147             final String projectID = info.getFirstValue(PROJECT_ID_PARAMETER, null);
148             final String defaultTimeUnit = DoiSettings.getInstance().getString(
149                     Consts.TOKEN_EXPIRATION_UNIT, 
150                     String.valueOf(TokenSecurity.TimeUnit.HOUR.getTimeUnit())
151             );
152             final String defaultTimeDelay = DoiSettings.getInstance().getString(
153                     Consts.TOKEN_EXPIRATION_DELAY, "1");
154             
155             final String timeParam = info.getFirstValue(UNIT_OF_TIME_PARAMETER, defaultTimeUnit);
156             final int timeUnit = Integer.parseInt(timeParam);
157             final TimeUnit unit = TokenSecurity.TimeUnit.getTimeUnitFrom(timeUnit);
158 
159             final int amount = Integer.parseInt(defaultTimeDelay);
160 
161             final String tokenJwt;
162             if (projectID == null) {
163                  tokenJwt = TokenSecurity.getInstance().generate(
164                         userID,
165                         unit,
166                         amount
167                 );                  
168             } else {
169                 tokenJwt = TokenSecurity.getInstance().generate(
170                         userID,
171                         Integer.parseInt(projectID),
172                         unit,
173                         amount
174                 );                
175             }
176 
177             LOG.info("Token created {} for project {} during {} {}",
178                     tokenJwt, projectID, amount, unit.name());
179 
180             if (this.tokenDB.addToken(tokenJwt)) {
181                 sendTokenToUser(user, userID, tokenJwt, amount, timeUnit);
182             }
183 
184             return LOG.traceExit(tokenJwt);
185         } catch (TokenSecurityException ex) {
186             throw LOG.throwing(Level.INFO, new ResourceException(ex.getStatus(), ex.getMessage(),
187                     ex));
188         }
189     }

191     /**
192      * Sends the token to the user when the administrator creates a token for theuser
193      *
194      * @param userAdmin User administration
195      * @param userID user to send the message
196      * @param token created token for userID
197      * @param amount time
198      * @param timeUnit time unit
199      */
200     private void sendTokenToUser(final String userAdmin, final String userID, final String token,
201             final int amount, final int timeUnit) {
202         if (!userAdmin.equals(userID)) {
203             try {
204                 final StringBuilder builderMsg = new StringBuilder();
205                 builderMsg.append("The administrator ").append(userAdmin)
206                         .append(" has created a token for you:\n\n").append(token).append("\n\n")
207                         .append("This token is valid during ").append(amount).append(" ")
208                         .append(TokenSecurity.TimeUnit.getTimeUnitFrom(timeUnit).name());
209                 final List<DOIUser> doiUsers = PluginFactory.getUserManagement().getUsers();
210                 String email = "";
211                 for (final DOIUser doiUser : doiUsers) {
212                     if (doiUser.getUsername().equals(userID)) {
213                         email = doiUser.getEmail();
214                         break;
215                     }
216                 }
217                 if (email.isEmpty()) {
218                     LOG.error("Email is not set for {}", userID);
219                 }
220                 EmailSettings.getInstance().sendMessage("Creating Token", builderMsg.toString(),
221                         email);
222             } catch (DOIDbException ex) {
223                 LOG.error(ex);
224             }
225         }
226     }

228     /**
229      * Returns the information from the token encoded as JSon format.
230      *
231      * @return the included information in the token
232      */
233     @Requirement(reqId = Requirement.DOI_SRV_180, reqName = Requirement.DOI_SRV_180_NAME)
234     @Get
235     public Representation getTokenInformation() {
236         LOG.traceEntry();
237         try {
238             final Jws<Claims> jws = TokenSecurity.getInstance()
239                     .getTokenInformation(this.tokenParam);
240             return LOG.traceExit(new JsonRepresentation(jws));
241         } catch (DoiRuntimeException ex) {
242             throw LOG.throwing(Level.INFO, new ResourceException(Status.CLIENT_ERROR_BAD_REQUEST,
243                     ex));
244         }
245     }
