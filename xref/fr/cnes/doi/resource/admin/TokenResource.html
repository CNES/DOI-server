116     /**
117      * Creates and stores a token.
118      *
119      * The token creation is based on several actions :
120      * <ul>
121      * <li>creates the {@link fr.cnes.doi.security.TokenSecurity#generate}</li>
122      * <li>stores the token in {@link fr.cnes.doi.db.AbstractTokenDBHelper token database}</li>
123      * </ul>
124      *
125      * @param infoForm submitted information when requesting the token creation
126      * @return the token
127      */
128     @Requirement(reqId = Requirement.DOI_SRV_150, reqName = Requirement.DOI_SRV_150_NAME)
129     @Post
130     public String createToken(final Form infoForm) {
131         LOG.traceEntry("Paramater : {}", infoForm);
132         final Form info = (infoForm == null) ? new Form() : infoForm;
133         try {
134             final String user = this.getClientInfo().getUser().getIdentifier();
135             LOG.debug("Identified user : {}", user);
136             final String userID;
137             if(isInRole(RoleAuthorizer.ROLE_ADMIN)) {
138                 // The admin can generate for everybody
139                 LOG.debug("User {} is admin", user);
140                 userID = info.getFirstValue(IDENTIFIER_PARAMETER, user);
141             } else {
142                 // The token is generated for the identified user.
143                 userID = user;
144             }
145             final String projectID = info.getFirstValue(PROJECT_ID_PARAMETER, null);
146             final String timeParam = info.getFirstValue(
147                     UNIT_OF_TIME_PARAMETER, String.
148                             valueOf(TokenSecurity.TimeUnit.HOUR.getTimeUnit())
149             );
150 
151             final int timeUnit = Integer.parseInt(timeParam);
152             final TimeUnit unit = TokenSecurity.TimeUnit.getTimeUnitFrom(timeUnit);
153 
154             final int amount = Integer.parseInt(info.getFirstValue(AMOUNT_OF_TIME_PARAMETER, "1"));
155 
156             final String tokenJwt;
157             if (projectID == null) {
158                  tokenJwt = TokenSecurity.getInstance().generate(
159                         userID,
160                         unit,
161                         amount
162                 );                  
163             } else {
164                 tokenJwt = TokenSecurity.getInstance().generate(
165                         userID,
166                         Integer.parseInt(projectID),
167                         unit,
168                         amount
169                 );                
170             }
171 
172             LOG.info("Token created {} for project {} during {} {}",
173                     tokenJwt, projectID, amount, unit.name());
174 
175             if (this.tokenDB.addToken(tokenJwt)) {
176                 sendTokenToUser(user, userID, tokenJwt, amount, timeUnit);
177             }
178 
179             return LOG.traceExit(tokenJwt);
180         } catch (TokenSecurityException ex) {
181             throw LOG.throwing(Level.INFO, new ResourceException(ex.getStatus(), ex.getMessage(),
182                     ex));
183         }
184     }

186     /**
187      * Sends the token to the user when the administrator creates a token for theuser
188      *
189      * @param userAdmin User administration
190      * @param userID user to send the message
191      * @param token created token for userID
192      * @param amount time
193      * @param timeUnit time unit
194      */
195     private void sendTokenToUser(final String userAdmin, final String userID, final String token,
196             final int amount, final int timeUnit) {
197         if (!userAdmin.equals(userID)) {
198             try {
199                 final StringBuilder builderMsg = new StringBuilder();
200                 builderMsg.append("The administrator ").append(userAdmin)
201                         .append(" has created a token for you:\n\n").append(token).append("\n\n")
202                         .append("This token is valid during ").append(amount).append(" ")
203                         .append(TokenSecurity.TimeUnit.getTimeUnitFrom(timeUnit).name());
204                 final List<DOIUser> doiUsers = PluginFactory.getUserManagement().getUsers();
205                 String email = "";
206                 for (final DOIUser doiUser : doiUsers) {
207                     if (doiUser.getUsername().equals(userID)) {
208                         email = doiUser.getEmail();
209                         break;
210                     }
211                 }
212                 if (email.isEmpty()) {
213                     LOG.error("Email is not set for {}", userID);
214                 }
215                 EmailSettings.getInstance().sendMessage("Creating Token", builderMsg.toString(),
216                         email);
217             } catch (DOIDbException ex) {
218                 LOG.error(ex);
219             }
220         }
221     }

223     /**
224      * Returns the information from the token encoded as JSon format.
225      *
226      * @return the included information in the token
227      */
228     @Requirement(reqId = Requirement.DOI_SRV_180, reqName = Requirement.DOI_SRV_180_NAME)
229     @Get
230     public Representation getTokenInformation() {
231         LOG.traceEntry();
232         try {
233             final Jws<Claims> jws = TokenSecurity.getInstance()
234                     .getTokenInformation(this.tokenParam);
235             return LOG.traceExit(new JsonRepresentation(jws));
236         } catch (DoiRuntimeException ex) {
237             throw LOG.throwing(Level.INFO, new ResourceException(Status.CLIENT_ERROR_BAD_REQUEST,
238                     ex));
239         }
240     }
