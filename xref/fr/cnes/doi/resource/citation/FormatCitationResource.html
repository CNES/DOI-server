34  /**
35   * Formats a citation. CrossRef, DataCite and mEDRA support formatted citations via the
36   * text/bibliography content type. These are the output of the Citation Style Language processor,
37   * citeproc-js. The content type can take two additional parameters to customise its response
38   * format.
39   * <p>
40   * "\"style\" can be chosen from the list of style names found in the CSL style repository. Many
41   * styles are supported, including common styles such as apa and harvard3
42   *
43   * @author Jean-Christophe Malapert (jean-christophe.malapert@cnes.fr)
44   */
45  public class FormatCitationResource extends BaseCitationResource {
46  
47      /**
48       * The digital object identifier to format.
49       */
50      private volatile String doiName;
51  
52      /**
53       * The style of formatting.
54       */
55      private volatile String style;
56  
57      /**
58       * The language to format.
59       */
60      private volatile String language;
61  
62      /**
63       * Init by getting doi, lang and style values.
64       *
65       * @throws ResourceException - if a problem happens
66       */
67      @Override
68      protected void doInit() throws ResourceException {
69          super.doInit();
70          LOG.traceEntry();
71          this.doiName = getQueryValue(DOI_PARAMETER);
72          this.language = getQueryValue(LANG_PARAMETER);
73          this.style = getQueryValue(STYLE_PARAMETER);
74  
75          LOG.debug("DOI Parameter : " + this.doiName);
76          LOG.debug("LANGUAGE Parameter : " + this.language);
77          LOG.debug("STYLE Parameter : " + this.language);
78  
79          final StringBuilder description = new StringBuilder();
80          description.append("CrossRef, DataCite and mEDRA support formatted "
81                  + "citations via the text/bibliography content type. These are "
82                  + "the output of the Citation Style Language processor, "
83                  + "citeproc-js. The content type can take two additional "
84                  + "parameters to customise its response format.");
85          description.append("\"style\" can be chosen from the list of style names"
86                  + " found in the CSL style repository. Many styles are supported,"
87                  + " including common styles such as apa and harvard3");
88          setDescription(description.toString());
89          LOG.traceExit();
90      }
91  
92      /**
93       * Returns the formatted citation.
94       *
95       * @return the formatted citation
96       * @throws ResourceException - if a problem happens when requesting Cross Cite or if
97       * {@link #DOI_PARAMETER} and {@link #LANG_PARAMETER} and {@link #STYLE_PARAMETER} are not set
98       */
99      @Requirement(reqId = Requirement.DOI_SRV_120, reqName = Requirement.DOI_SRV_120_NAME)
100     @Requirement(reqId = Requirement.DOI_MONIT_020, reqName = Requirement.DOI_MONIT_020_NAME)
101     @Get
102     public String getFormat() throws ResourceException {
103         LOG.traceEntry();
104         try {
105             checkInputs();
106             final String result = this.getApp().getClient().getFormat(
107                     this.doiName, this.style, this.language
108             );
109             return LOG.traceExit(result);
110         } catch (ClientCrossCiteException ex) {
111             ((AbstractApplication) getApplication()).sendAlertWhenDataCiteFailed(ex);
112             throw LOG.throwing(Level.ERROR, new ResourceException(ex.getStatus(), ex.
113                     getDetailMessage(), ex));
114         }
115     }
116 
117     /**
118      * Checks input parameters
119      *
120      * @throws ResourceException - if {@link #DOI_PARAMETER} and {@link #LANG_PARAMETER} and
121      * {@link #STYLE_PARAMETER} are not set
122      */
123     private void checkInputs() throws ResourceException {
124         LOG.traceEntry();
125         final StringBuilder errorMsg = new StringBuilder();
126         if (this.doiName == null || this.doiName.isEmpty()) {
127             errorMsg.append(DOI_PARAMETER).append(" value is not set.");
128         }
129         if (this.language == null || this.language.isEmpty()) {
130             errorMsg.append(LANG_PARAMETER).append(" value is not set.");
131         }
132         if (this.style == null || this.style.isEmpty()) {
133             errorMsg.append(STYLE_PARAMETER).append(" value is not set.");
134         }
135         if (errorMsg.length() == 0) {
136             LOG.debug("The parameters are valid");
137         } else {
138             throw LOG.throwing(Level.ERROR, new ResourceException(
139                     Status.CLIENT_ERROR_BAD_REQUEST, errorMsg.toString()));
140         }
141         LOG.traceExit();
142     }

144     /**
145      * Describes the Get Method.
146      *
147      * @param info Wadl description
148      */
149     @Requirement(reqId = Requirement.DOI_DOC_010, reqName = Requirement.DOI_DOC_010_NAME)
150     @Override
151     protected final void describeGet(final MethodInfo info) {
152         info.setName(Method.GET);
153         info.setDocumentation("Select Formatting Style");
154         addRequestDocToMethod(info, Arrays.asList(
155                 createQueryParamDoc(DOI_PARAMETER, ParameterStyle.QUERY,
156                         "doi to format for the citation", true, "xs:string"),
157                 createQueryParamDoc(LANG_PARAMETER, ParameterStyle.QUERY,
158                         "language for the citation formating", true, "xs:string"),
159                 createQueryParamDoc(STYLE_PARAMETER, ParameterStyle.QUERY,
160                         "style fot the citation formating", true, "xs:string")
161         ));
162         addResponseDocToMethod(info, createResponseDoc(
163                 Status.SUCCESS_OK,
164                 "Operation successful",
165                 listRepresentation("Format representation", MediaType.TEXT_PLAIN,
166                         "The formatted citation"))
167         );
168         addResponseDocToMethod(info, createResponseDoc(
169                 Status.CLIENT_ERROR_NOT_FOUND,
170                 "DOI not found",
171                 listRepresentation("Error representation",
172                         MediaType.TEXT_HTML, "Error"))
173         );
174         addResponseDocToMethod(info, createResponseDoc(
175                 Status.CLIENT_ERROR_BAD_REQUEST,
176                 "Wrong input parameters",
177                 listRepresentation("Error representation",
178                         MediaType.TEXT_HTML, "Error"))
179         );
180     }
