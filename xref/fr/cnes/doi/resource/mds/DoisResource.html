74      /**
75       * Returns the collection of DOI. This request returns a list of all DOIs for the requesting
76       * datacentre. Status 200 is returned when operation is successful.
77       *
78       * @return the list of DOI
79       * @throws DoiServerException 204 No Content - no DOIs founds
80       */
81      @Get
82      public List<String> getDois() throws DoiServerException {
83          LOG.traceEntry();
84          setStatus(Status.SUCCESS_OK);
85          final List<String> dois = this.getDoiApp().getClientSearch().getDois();
86          if (dois == null || dois.isEmpty()) {
87              setStatus(Status.SUCCESS_NO_CONTENT);
88          } else {
89              setStatus(Status.SUCCESS_OK);
90          }
91          return LOG.traceExit(dois);
92      }

94      /**
95       * Creates a new DOI based on the doi and url parameters
96       *
97       * The DOI name is built following this syntax:
98       * <i>CNES_prefix</i>/<i>project_provided_the_DOI_server</i>/project_suffix.
99       *
100      * <p>
101      * The client provides :
102      * <ul>
103      * <li><i>doi</i> : the project suffix, which is an unique identifier within the project.</li>
104      * <li><i>url</i> : the URL of the landing page.
105      * </ul>
106      * <b>Note 1</b> : the landing page should be accessible from www
107      * <b>Note 2</b> : Metadata must be uploaded first
108      *
109      * <p>
110      * Will mint new DOI if specified DOI doesn't exist. This method will attempt to update URL if
111      * you specify existing DOI. Standard domains and quota restrictions check will be performed. A
112      * Datacentre's doiQuotaUsed will be increased by 1. A new record in Datasets will be created by
113      * returning a 201 status for operation successful.
114      *
115      * @param doiForm doi and url
116      * @return short explanation of status code e.g. CREATED, HANDLE_ALREADY_EXISTS etc
117      * @throws DoiServerException - if the response is not a success
118      * <ul>
119      * <li>{@link DATACITE_API_RESPONSE#PROCESS_ERROR}</li>
120      * <li>{@link API_MDS#DATACITE_PROBLEM}</li>
121      * <li>{@link API_MDS#SECURITY_USER_NO_ROLE}</li>
122      * <li>{@link API_MDS#SECURITY_USER_NOT_IN_SELECTED_ROLE}</li>
123      * <li>{@link API_MDS#SECURITY_USER_PERMISSION}</li>
124      * <li>{@link API_MDS#SECURITY_USER_CONFLICT}</li>
125      * </ul>
126      */
127     @Requirement(reqId = Requirement.DOI_SRV_020, reqName = Requirement.DOI_SRV_020_NAME)
128     @Requirement(reqId = Requirement.DOI_SRV_030, reqName = Requirement.DOI_SRV_030_NAME)
129     @Requirement(reqId = Requirement.DOI_MONIT_020, reqName = Requirement.DOI_MONIT_020_NAME)
130     @Requirement(reqId = Requirement.DOI_INTER_070, reqName = Requirement.DOI_INTER_070_NAME)
131     @Requirement(reqId = Requirement.DOI_AUTO_020, reqName = Requirement.DOI_AUTO_020_NAME)
132     @Requirement(reqId = Requirement.DOI_AUTO_030, reqName = Requirement.DOI_AUTO_030_NAME)
133     @Post("form")
134     public String createDoi(final Form doiForm) throws DoiServerException {
135         LOG.traceEntry("Parameters\n\tdoiForm : {}", doiForm);
136         checkInputs(doiForm);
137         final String result;
138         final String selectedRole = extractSelectedRoleFromRequestIfExists();
139         checkPermission(doiForm.getFirstValue(DOI_PARAMETER), selectedRole);
140         try {
141             setStatus(Status.SUCCESS_CREATED, "Operation successful");
142             result = this.getDoiApp().getClient().createDoi(doiForm);
143         } catch (ClientMdsException ex) {
144             if (ex.getStatus().getCode() == Status.CLIENT_ERROR_PRECONDITION_FAILED.getCode()) {
145                 throw LOG.throwing(
146                         Level.ERROR,
147                         new DoiServerException(getApplication(), DATACITE_API_RESPONSE.PROCESS_ERROR,
148                                 ex)
149                 );
150             } else {
151                 throw LOG.throwing(
152                         Level.ERROR,
153                         new DoiServerException(getApplication(), API_MDS.DATACITE_PROBLEM, ex)
154                 );
155             }
156         }
157 
158         return LOG.traceExit(result);
159     }

161     /**
162      * Checks input parameters. Checks that
163      * {@value fr.cnes.doi.resource.mds.BaseMdsResource#DOI_PARAMETER} and
164      * {@value fr.cnes.doi.resource.mds.BaseMdsResource#URL_PARAMETER} are provided in the
165      * mediaForm.
166      *
167      * @param mediaForm the parameters
168      * {@value fr.cnes.doi.resource.mds.BaseMdsResource#DOI_PARAMETER} and
169      * {@value fr.cnes.doi.resource.mds.BaseMdsResource#URL_PARAMETER}
170      * @throws DoiServerException - 400 Bad Request if DOI_PARAMETER and
171      * {@value fr.cnes.doi.resource.mds.BaseMdsResource#URL_PARAMETER} are not set
172      */
173     @Requirement(reqId = Requirement.DOI_INTER_070, reqName = Requirement.DOI_INTER_070_NAME)
174     private void checkInputs(final Form mediaForm) throws DoiServerException {
175         LOG.traceEntry("Parameter : {}", mediaForm);
176         StringBuilder errorMsg = new StringBuilder();
177         if (isValueNotExist(mediaForm, DOI_PARAMETER)) {
178             errorMsg = errorMsg.append(DOI_PARAMETER).append(" value is not set.");
179         } else {
180             try {
181                 ClientMDS.checkIfAllCharsAreValid(mediaForm.getFirstValue(DOI_PARAMETER));
182             } catch (IllegalArgumentException ex) {
183                 errorMsg = errorMsg.append(DOI_PARAMETER).append(" no valid syntax.");
184             }
185         }
186         if (isValueNotExist(mediaForm, URL_PARAMETER)) {
187             errorMsg = errorMsg.append(URL_PARAMETER).append(" value is not set.");
188         }
189         if (errorMsg.length() == 0) {
190             LOG.debug("The form is valid");
191         } else {
192             throw LOG.throwing(
193                     Level.ERROR,
194                     new DoiServerException(getApplication(), API_MDS.LANGING_PAGE_VALIDATION,
195                             errorMsg.toString())
196             );
197         }
198         LOG.traceExit();
199     }
