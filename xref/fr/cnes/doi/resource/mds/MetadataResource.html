78      /**
79       * Checks input parameters
80       *
81       * @param doiName DOI number
82       * @throws DoiServerException - 400 Bad Request if DOI_PARAMETER is not set
83       */
84      @Requirement(reqId = Requirement.DOI_INTER_070, reqName = Requirement.DOI_INTER_070_NAME)
85      private void checkInputs(final String doiName) throws DoiServerException {
86          LOG.traceEntry("Parameter\n\tdoiName : {}", doiName);
87          StringBuilder errorMsg = new StringBuilder();
88          if (doiName == null || doiName.isEmpty()) {
89              errorMsg = errorMsg.append(DoiMdsApplication.DOI_TEMPLATE).append("value is not set.");
90          } else {
91              try {
92                  ClientMDS.checkIfAllCharsAreValid(doiName);
93              } catch (IllegalArgumentException ex) {
94                  errorMsg = errorMsg.append(DoiMdsApplication.DOI_TEMPLATE).
95                          append("no valid syntax.");
96              }
97          }
98          if (errorMsg.length() == 0) {
99              LOG.debug("The input is valid");
100         } else {
101             throw LOG.throwing(
102                     Level.ERROR,
103                     new DoiServerException(getApplication(), Status.CLIENT_ERROR_BAD_REQUEST,
104                             errorMsg.toString())
105             );
106         }
107         LOG.traceExit();
108     }

110     /**
111      * Retuns the metadata for a given DOI. 200 status is returned when the operation is successful.
112      *
113      * @return the metadata for a given DOI as Json or XML
114      * @throws DoiServerException - if the response is not a success
115      * <ul>
116      * <li>{@link DATACITE_API_RESPONSE#BAD_REQUEST}</li>
117      * <li>{@link DATACITE_API_RESPONSE#DOI_NOT_FOUND}</li>
118      * <li>{@link API_MDS#DATACITE_PROBLEM}</li>
119      * </ul>
120      */
121     @Requirement(reqId = Requirement.DOI_SRV_060, reqName = Requirement.DOI_SRV_060_NAME)
122     @Requirement(reqId = Requirement.DOI_MONIT_020, reqName = Requirement.DOI_MONIT_020_NAME)
123     @Get("xml")
124     public Representation getMetadata() throws DoiServerException {
125         LOG.traceEntry();
126         checkInputs(doiName);
127         final Representation resource;
128         try {
129             setStatus(Status.SUCCESS_OK);
130             resource = this.getDoiApp().getClient().getMetadata(this.doiName);
131         } catch (ClientMdsException ex) {
132             if (ex.getStatus().getCode() == Status.CLIENT_ERROR_NOT_FOUND.getCode()) {
133                 throw LOG.throwing(
134                         Level.ERROR,
135                         new DoiServerException(getApplication(), DATACITE_API_RESPONSE.DOI_NOT_FOUND,
136                                 ex)
137                 );
138             } else if (ex.getStatus().getCode() == Status.CLIENT_ERROR_BAD_REQUEST.getCode()) {
139                 throw LOG.throwing(
140                         Level.ERROR,
141                         new DoiServerException(getApplication(), DATACITE_API_RESPONSE.BAD_REQUEST,
142                                 ex)
143                 );
144             } else {
145                 throw LOG.throwing(
146                         Level.ERROR,
147                         new DoiServerException(getApplication(), API_MDS.DATACITE_PROBLEM, ex)
148                 );
149             }
150         }
151         return LOG.traceExit(resource);
152     }

167     /**
168      * Deletes a representation for a given DOI. 200 status when the operation is successful.
169      *
170      * @return the deleted representation
171      * @throws DoiServerException - if the response is not a success
172      * <ul>

174      * <li>{@link API_MDS#SECURITY_USER_NOT_IN_SELECTED_ROLE}</li>
175      * <li>{@link API_MDS#SECURITY_USER_PERMISSION}</li>

177      * <li>{@link DATACITE_API_RESPONSE#DOI_NOT_FOUND}</li>
178      * <li>{@link API_MDS#DATACITE_PROBLEM}</li>
179      * </ul>
180      */
181     @Requirement(reqId = Requirement.DOI_SRV_050, reqName = Requirement.DOI_SRV_050_NAME)
182     @Requirement(reqId = Requirement.DOI_MONIT_020, reqName = Requirement.DOI_MONIT_020_NAME)
183     @Requirement(reqId = Requirement.DOI_INTER_070, reqName = Requirement.DOI_INTER_070_NAME)
184     @Requirement(reqId = Requirement.DOI_AUTO_020, reqName = Requirement.DOI_AUTO_020_NAME)
185     @Requirement(reqId = Requirement.DOI_AUTO_030, reqName = Requirement.DOI_AUTO_030_NAME)
186     @Delete
187     public Representation deleteMetadata() throws DoiServerException {
188         LOG.traceEntry();
189         checkInputs(this.doiName);
190         final Representation rep;
191         try {
192             final String selectedRole = extractSelectedRoleFromRequestIfExists();
193             checkPermission(this.doiName, selectedRole);
194             setStatus(Status.SUCCESS_OK);
195             rep = this.getDoiApp().getClient().deleteMetadata(this.doiName);
196         } catch (ClientMdsException ex) {
197             if (ex.getStatus().getCode() == Status.CLIENT_ERROR_NOT_FOUND.getCode()) {
198                 throw LOG.throwing(
199                         Level.ERROR,
200                         new DoiServerException(getApplication(), DATACITE_API_RESPONSE.DOI_NOT_FOUND,
201                                 ex)
202                 );
203             } else {
204                 throw LOG.throwing(
205                         Level.ERROR,
206                         new DoiServerException(getApplication(), API_MDS.DATACITE_PROBLEM, ex)
207                 );
208             }
209         }
210         return LOG.traceExit(rep);
211     }
