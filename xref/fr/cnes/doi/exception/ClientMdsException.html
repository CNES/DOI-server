27   * Exception for Client Cross Cite.

29   * @author Jean-Christophe Malapert (jean-christophe.malapert@cnes.fr)

31  public class ClientMdsException extends Exception {
32  
33      /**
34       * Operation successful. 201
35       */
36      public static final int SUCCESS_CREATED = 201;
37  
38      /**
39       * Operation successful. 200
40       */
41      public static final int SUCCESS_OK = 200;
42  
43      /**
44       * no DOIs founds. 204
45       */
46      public static final int SUCCESS_NO_CONTENT = 204;
47  
48      /**
49       * invalid XML, wrong prefix or request body must be exactly two lines: "DOI and URL; wrong
50       * domain, wrong prefix". 400
51       */
52      public static final int CLIENT_BAD_REQUEST = 400;
53  
54      /**
55       * no login. 401
56       */
57      public static final int CLIENT_ERROR_UNAUTHORIZED = 401;
58  
59      /**
60       * no login. 1001
61       */
62      public static final int CONNECTOR_ERROR_COMMUNICATION = 1001;
63  
64      /**
65       * login problem, quota exceeded or dataset belongs to another party. 403
66       */
67      public static final int CLIENT_ERROR_FORBIDDEN = 403;
68  
69      /**
70       * DOI does not exist in our database. 404
71       */
72      public static final int CLIENT_ERROR_NOT_FOUND = 404;
73  
74      /**
75       * DOIServer : Cannot know which role must be applied. 409
76       */
77      public static final int CLIENT_ERROR_CONFLICT = 409;
78  
79      /**
80       * the requested dataset was marked inactive (using DELETE method). 410
81       */
82      public static final int CLIENT_ERROR_GONE = 410;
83  
84      /**
85       * metadata must be uploaded first. 412
86       */
87      public static final int CLIENT_ERROR_PRECONDITION_FAILED = 412;
88  
89      /**
90       * SeralVersionUID
91       */
92      private static final long serialVersionUID = -5061913391706889102L;
93  
94      /**
95       * Detail message.
96       */
97      private final String detailMessage;
98  
99      /**
100      * HTTP status.
101      */
102     private final Status status;
103 
104     /**
105      * Constructs a new exception with HTTP status as its detail message.
106      *
107      * @param status HTTP status
108      */
109     public ClientMdsException(final Status status) {
110         super();
111         this.detailMessage = computeDetailMessage(status);
112         this.status = computeStatus(status);
113     }
114 
115     /**
116      * Constructs a new exception with the specified HTTP status and detail message.
117      *
118      * @param status HTTP message
119      * @param message message
120      */
121     public ClientMdsException(final Status status,
122             final String message) {
123         super(message);
124         this.detailMessage = computeDetailMessage(status);
125         this.status = computeStatus(status);
126     }
127 
128     /**
129      * Constructs a new exception with the specified detail cause.
130      *
131      * @param status HTTP status
132      * @param cause cause
133      */
134     public ClientMdsException(final Status status,
135             final Throwable cause) {
136         super(cause);
137         this.detailMessage = computeDetailMessage(status);
138         this.status = computeStatus(status);
139     }
140 
141     /**
142      * Constructs a new exception with the specified detail HTTP status, message and cause.
143      *
144      * @param status HTTP status
145      * @param message message
146      * @param cause cause
147      */
148     public ClientMdsException(final Status status,
149             final String message,
150             final Throwable cause) {
151         super(message, cause);
152         this.detailMessage = computeDetailMessage(status);
153         this.status = computeStatus(status);
154     }
155 
156     /**
157      * Constructs a new exception with the specified detail HTTP status, response and cause.
158      *
159      * @param status HTTP status
160      * @param message message
161      * @param responseEntity Representation of the response
162      * @param cause cause
163      */
164     public ClientMdsException(final Status status,
165             final String message,
166             final Representation responseEntity,
167             final Throwable cause) {
168         super(message, cause);
169         this.status = computeStatus(status);
170         String txt;
171         try {
172             txt = responseEntity.getText();
173         } catch (IOException ex) {
174             txt = computeDetailMessage(this.status);
175         }
176         this.detailMessage = txt;
177     }
178 
179     /**
180      * Computes status
181      *
182      * @param status status
183      * @return status
184      */
185     private Status computeStatus(final Status status) {
186         return (status.getCode() == CONNECTOR_ERROR_COMMUNICATION) ? new Status(
187                 CLIENT_ERROR_UNAUTHORIZED) : status;
188     }
189 
190     /**
191      * Returns the detail message according to the status.
192      *
193      * @param status HTTP status
194      * @return the detail message
195      */
196     private String computeDetailMessage(final Status status) {
197         return DATACITE_API_RESPONSE.getMessageFromStatus(status);
198     }
199 
200     /**
201      * Returns the status.
202      *
203      * @return the status
204      */
205     public Status getStatus() {
206         return this.status;
207     }
208 
209     /**
210      * Returns detail message;
211      *
212      * @return the detail message
213      */
214     public String getDetailMessage() {
215         return this.detailMessage;
216     }
