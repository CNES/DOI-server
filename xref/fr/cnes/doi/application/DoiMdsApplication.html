59  /**
60   * Provides an application to handle Data Object Identifier within an
61   * organization. A Digital Object Identifier (DOI) is a persistent identifier or
62   * handle used to uniquely identify objects, standardized by the International
63   * Organization. A DOI aims to be "resolvable", usually to some form of access
64   * to the information object to which the DOI refers. This is achieved by
65   * binding the DOI to metadata about the object, such as a URL, where all the
66   * details about the object are accessible. Everytime a URL changes, the
67   * publisher has to update the metadata for the DOI to link to the new URL. It
68   * is the publisher's responsibility to update the DOI database. If he fails to
69   * do so, the DOI resolves to a dead link leaving the DOI useless.
70   * <p>
71   * Two methods of authentication are defined in this application and both is
72   * optional:
73   * <ul>
74   * <li>{@link #createAuthenticator authenticator} by login/password</li>
75   * <li>{@link #createTokenAuthenticator authenticator} by
76   * {@link fr.cnes.doi.resource.admin.TokenResource token}
77   * </ul>
78   * Only the GET can be anonymous whereas POST, PUT, DELETE
79   * {@link #createMethodAuthorizer methods need to be authenticated}.
80   *
81   * <p>
82   * <b>Security</b><br>
83   * <code>
84   * --------------<br>
85   * The authentication is done by the following pipeline:<br>
86   * |Method authorization|--&gt;|Authentication login/pwd|--&gt;|Authentication
87   * token|<br>
88   * Method authorization : Only GET method does not need an authorization
89   * </code>
90   *
91   * <p>
92   * <b>Routing</b><br>
93   * <code>
94   * --------------<br>
95   * <br>
96   * root<br>
97   * |<br>
98   * |__ dois (authorization)<br>
99   * |__ dois/{doiName} (authorization)<br>
100  * |__ metadata (authorization)<br>
101  * |__ metadata/{doiName} (authorization)<br>
102  * |__ media/{doiName} (authorization)<br>
103  * </code>
104  *
105  * @see #createInboundRoot the resources related to this application
106  * @see <a href="http://www.doi.org/hb.html">DOI Handbook</a>
107  * @see <a href="https://mds.datacite.org/static/apidoc">API Documentation</a>
108  * @see DoisResource List and create DOI
109  * @see DoiResource Handle a DOI
110  * @see MetadatasResource Create metadata
111  * @see MetadataResource Handle DOI metadata
112  * @see MediaResource Handle media related to metadata
113  *
114  * @author Jean-Christophe Malapert (jean-Christophe Malapert@cnes.fr)
115  */

204     /**
205      * Creates a router for the DoiMdsApplication.
206      *
207      * This router routes the resources for the Mds application, which is
208      * protected by two authentication mechanisms (optional mechanisms) and an
209      * authorization by method.
210      *
211      * @see DoiMdsApplication#createRouter the router that contains the Mds
212      * resources
213      * @see DoiMdsApplication#createAuthenticator the authentication mechanism
214      * by login/password
215      * @see DoiMdsApplication#createTokenAuthenticator the authentication
216      * mechanism by token
217      * @see DoiMdsApplication#createMethodAuthorizer the method authorization
218      * mechanism
219      * @return Router
220      */
221     @Override
222     public Restlet createInboundRoot() {
223         LOG.traceEntry();
224 
225         final Filter logContext = new Filter() {
226             @Override
227             protected int beforeHandle(Request request, Response response) {
228                 final ClientInfo clientInfo = request.getClientInfo();
229                 final String ipAddress = request.getHeaders().getFirstValue(
230                         Consts.PROXIFIED_IP, clientInfo.getUpstreamAddress()
231                 );
232                 ThreadContext.put(Consts.LOG_IP_ADDRESS, ipAddress);
233                 return Filter.CONTINUE;
234             }
235         };        
236 
237         // Defines the strategy of authentication (authentication is not required)
238         //   - authentication with login/pwd
239         final ChallengeAuthenticator challAuth = createAuthenticatorLoginBased();
240         challAuth.setOptional(true);
241 
242         //   - authentication with token
243         final ChallengeAuthenticator challTokenAuth = createTokenAuthenticator();
244         challTokenAuth.setOptional(true);
245 
246         // Set specific authorization on method after checking authentication
247         final MethodAuthorizer methodAuth = createMethodAuthorizer();
248 
249         // set information available for Log4j
250         logContext.setNext(challAuth);
251         
252         //  create a pipeline of authentication
253         challAuth.setNext(challTokenAuth);
254         challTokenAuth.setNext(methodAuth);
255 
256         // Router
257         methodAuth.setNext(createRouter());
258 
259         final Filter filter = new SecurityPostProcessingFilter(getContext(), logContext);
260         return LOG.traceExit(filter);
261     }

263     /**
264      * Creates the router. The router routes the following resources:
265      * <ul>
266      * <li>{@link DoiMdsApplication#DOI_URI} to create/update a DOI and its
267      * landing page</li>
268      * <li>{@link DoiMdsApplication#DOI_URI} {@link DoiMdsApplication#DOI_NAME_URI}
269      * to get the URL of the landing page related to a given DOI</li>
270      * <li>{@link DoiMdsApplication#METADATAS_URI} to create/update DOI
271      * metadata</li>
272      * <li>{@link DoiMdsApplication#METADATAS_URI} {@link DoiMdsApplication#DOI_NAME_URI}
273      * to get DOI's metadata or delete a given DOI</li>
274      * <li>{@link DoiMdsApplication#MEDIA_URI} {@link DoiMdsApplication#DOI_NAME_URI}
275      * to handle media related to a DOI
276      * </ul>
277      *
278      * @see DoiResource Handles a DOI and its landing page
279      * @see MetadatasResource Handles DOI metadata
280      * @see MetadataResource Handles DOI metadata
281      * @see MediaResource Handles media related to a DOI
282      * @return the router
283      */
284     private Router createRouter() {
285         LOG.traceEntry();
286 
287         final Router router = new Router(getContext());
288         router.attach(DOI_URI, DoisResource.class);
289         router.attach(DOI_URI + DOI_NAME_URI, DoiResource.class)
290                 .getTemplate().setMatchingMode(Template.MODE_STARTS_WITH);
291         router.attach(METADATAS_URI, MetadatasResource.class);
292         router.attach(METADATAS_URI + DOI_NAME_URI, MetadataResource.class)
293                 .getTemplate().setMatchingMode(Template.MODE_STARTS_WITH);
294         router.attach(MEDIA_URI + DOI_NAME_URI, MediaResource.class)
295                 .getTemplate().setMatchingMode(Template.MODE_STARTS_WITH);
296 
297         return LOG.traceExit(router);
298     }

300     /**
301      * Creates the method authorizer. GET method can be anonymous. The verbs
302      * (POST, PUT, DELETE) need to be authenticated.
303      *
304      * @return Authorizer based on authorized methods
305      */
306     private MethodAuthorizer createMethodAuthorizer() {
307         LOG.traceEntry();
308 
309         final MethodAuthorizer methodAuth = new MethodAuthorizer();
310         methodAuth.getAnonymousMethods().add(Method.GET);
311         methodAuth.getAnonymousMethods().add(Method.OPTIONS);
312         methodAuth.getAuthenticatedMethods().add(Method.GET);
313         methodAuth.getAuthenticatedMethods().add(Method.POST);
314         methodAuth.getAuthenticatedMethods().add(Method.PUT);
315         methodAuth.getAuthenticatedMethods().add(Method.DELETE);
316 
317         return LOG.traceExit(methodAuth);
318     }

441         @Override
442         protected void afterHandle(final Request request, final Response response) {
443             final Status status = response.getStatus();
444             final String reason = status.getReasonPhrase();
445             if (status.getCode() == Status.CLIENT_ERROR_UNAUTHORIZED.getCode()
446                     && API_MDS.SECURITY_USER_NO_ROLE.getShortMessage().equals(reason)) {

448                         "Basic realm=\"DOI Server access\", charset=\"UTF-8\"");

457     public enum API_MDS {
458         /**
459          * Create metadata. SUCCESS_CREATED is used as status meaning "Operation

468         SECURITY_USER_NOT_IN_SELECTED_ROLE(Status.CLIENT_ERROR_FORBIDDEN,

470         /**
471          * Fail to authorize the user. It happens when a client is authentified
472          * but unauthorized to use the resource. CLIENT_ERROR_UNAUTHORIZED is
473          * used as status meaning "Fail to authorize the user"
474          */
475         SECURITY_USER_NO_ROLE(Status.CLIENT_ERROR_UNAUTHORIZED, "Fail to authorize the user"),
476         /**
477          * Fail to know privileges of a user. It happens when an user is
478          * associated to several roles without selecting one.
479          * CLIENT_ERROR_CONFLICT is used as status meaning "Error when an user
480          * is associated to more than one role without setting selectedRole
481          * parameter"
482          */
483         SECURITY_USER_CONFLICT(Status.CLIENT_ERROR_CONFLICT,
484                 "Error when an user is associated to more than one role without setting selectedRole "
485                 + "parameter"),
486         /**
487          * Fail to create a DOI. It happens when a user try to create a DOI with
488          * the wrong role. Actually, the role is contained in the DOI name. So
489          * if a user is authentified with a right role and try to create a DOI
490          * for another role, an exception is raised. SECURITY_USER_PERMISSION is
491          * used as status meaning "User is not allowed to make this operation"
492          */
493         SECURITY_USER_PERMISSION(Status.CLIENT_ERROR_FORBIDDEN,
494                 "User is not allowed to make this operation"),
495         /**
496          * Cannot access to Datacite. It happens when a network problem may
497          * happen with Datacite. SERVER_ERROR_GATEWAY_TIMEOUT is used as status
498          * meaning "Cannot access to Datacite"
499          */
500         NETWORK_PROBLEM(Status.CONNECTOR_ERROR_COMMUNICATION, "Cannot access to Datacite"),
501         /**
502          * Fail to validate user input parameter for creating the DOI. It
503          * happens in the following cases:
504          * <ul>
505          * <li>the DOI or metadata are not provided</li>
506          * <li>the prefix is not allowed</li>
507          * <li>some characters are not allowed in the DOI name</li>
508          * </ul>
509          * CLIENT_ERROR_BAD_REQUEST is used as status meaning "Failed to

511          */
512         METADATA_VALIDATION(Status.CLIENT_ERROR_BAD_REQUEST,

514         /**

521                 "DOI not provided or one or more of the specified mime-types or urls are invalid "
522                 + "(e.g. non supported mime-type, not allowed url domain, etc.)"),
523         /**
524          * Fail to create the landing page related to the DOI.
525          * CLIENT_ERROR_BAD_REQUEST is used as status meaning "Validation error
526          * when defining the DOI and its landing page"
527          */
528         LANGING_PAGE_VALIDATION(Status.CLIENT_ERROR_BAD_REQUEST,

530         /**
