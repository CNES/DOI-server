47  /**
48   * Creates a base application by setting both the CORS, the DOI status page, the proxy and the DOI.

55      /**
56       * Default value of 'Access-Control-Allow-Origin' header.
57       */

61  
62      /**
63       * If true, adds 'Access-Control-Allow-Credentials' header.

67      /**

70      private static final Logger LOG = LogManager.getLogger(AbstractApplication.class.getName());

78      private final DoiSettings config = DoiSettings.getInstance();

80      /**
81       * This constructor creates an instance of proxySettings and doiSettings. By creating the
82       * instance, the constructor creates :
83       * <ul>
84       * <li>the CORS service using the default value {@link AbstractApplication#DEFAULT_CORS_ORIGIN}
85       * and {@link AbstractApplication#DEFAULT_CORS_CREDENTIALS}</li>
86       * <li>the custom status page</li>
87       * </ul>

98       * Defined services and metadata.
99       */
100     private void init() {
101         getServices().add(this.createCoreService(DEFAULT_CORS_ORIGIN, DEFAULT_CORS_CREDENTIALS));
102         setStatusService(new CnesStatusService());

104         setAuthor("Jean-Christophe Malapert (DNO/ISA/VIP)");

107     /**
108      * Defines the CORS service.
109      *
110      * @param corsOrigin IP that can access to the service
111      * @param corsCredentials credentials allowed
112      * @return CORS service
113      */
114     protected final CorsService createCoreService(final Set corsOrigin,
115             final boolean corsCredentials) {
116         LOG.traceEntry();
117         final CorsService corsService = new CorsService();
118         LOG.info("Allows all origins {}", corsOrigin);
119         corsService.setAllowedOrigins(corsOrigin);
120         LOG.info("Allows Credientials {}", corsCredentials);
121         corsService.setAllowedCredentials(corsCredentials);
122 
123         return LOG.traceExit(corsService);
124     }

126     /**
127      * Creates the authenticator based on a HTTP basic. Creates the user, role and mapping
128      * user/role.
129      *
130      * @return Authenticator based on a challenge scheme
131      */
132     @Requirement(reqId = Requirement.DOI_AUTH_010, reqName = Requirement.DOI_AUTH_010_NAME)
133     protected ChallengeAuthenticator createAuthenticator() {
134         LOG.traceEntry();
135         final ChallengeAuthenticator guard = new ChallengeAuthenticator(
136                 getContext(), ChallengeScheme.HTTP_BASIC, "realm");
137 
138         guard.setVerifier(this.getContext().getDefaultVerifier());
139         guard.setEnroler(this.getContext().getDefaultEnroler());
140 
141         return LOG.traceExit(guard);
142     }

144     /**
145      * Creates the authenticator based on a HTTP basic. Creates the user, role and mapping
146      * user/role.
147      *
148      * @return Authenticator based on a challenge scheme
149      */
150     @Requirement(reqId = Requirement.DOI_AUTH_010, reqName = Requirement.DOI_AUTH_010_NAME)
151     protected ChallengeAuthenticator createAuthenticatorLoginBased() {
152         LOG.traceEntry();
153         final ChallengeAuthenticator guard = new ChallengeAuthenticator(
154                 getContext(), ChallengeScheme.HTTP_BASIC, "realm") {
155             /**
156              * Cancel verification on pre-flight OPTIONS method
157              *
158              * @param request request
159              * @param response response
160              * @return the result
161              */
162             @Override
163             public int beforeHandle(final Request request, final Response response) {
164                 final int status;
165                 if (request.getMethod().equals(Method.OPTIONS)) {
166                     response.setStatus(Status.SUCCESS_OK);
167                     status = org.restlet.routing.Filter.CONTINUE;
168                 } else {
169                     status = super.beforeHandle(request, response);
170                 }
171                 return status;
172             }
173         };
174 
175         final LoginBasedVerifier verifier = new LoginBasedVerifier();
176         guard.setVerifier(verifier);
177 
178         return LOG.traceExit(guard);
179     }

181     /**
182      * Creates an authentication by token.
183      *
184      * @return the object that contains the business to check
185      * @see TokenBasedVerifier the token verification
186      */
187     @Requirement(reqId = Requirement.DOI_AUTH_020, reqName = Requirement.DOI_AUTH_020_NAME)
188     protected ChallengeAuthenticator createTokenAuthenticator() {
189         LOG.traceEntry();
190         final ChallengeAuthenticator guard = new ChallengeAuthenticator(
191                 getContext(), ChallengeScheme.HTTP_OAUTH_BEARER, "testRealm") {
192             /**
193              * Verifies the token.
194              *
195              * @param request request
196              * @param response response
197              * @return the result
198              */
199             @Override
200             public int beforeHandle(final Request request, final Response response) {
201                 final int status;
202                 if (request.getMethod().equals(Method.OPTIONS)) {
203                     response.setStatus(Status.SUCCESS_OK);
204                     status = org.restlet.routing.Filter.CONTINUE;
205                 } else {
206                     status = super.beforeHandle(request, response);
207                 }
208                 return status;
209             }
210         };
211         final TokenBasedVerifier verifier = new TokenBasedVerifier(getTokenDB());
212         guard.setVerifier(verifier);
213 
214         return LOG.traceExit(guard);
215     }

229     /**
230      * Sends alert as an email when DataCite failed.
231      *
232      * @param exception error message to send
233      */
234     public void sendAlertWhenDataCiteFailed(final Exception exception) {
235         LOG.traceEntry("Parameters : {}", exception);
236         final String subject = "Datacite problem";
237         final String message = "Dear administrator, an error has been detected"
238                 + " coming from Datacite, please look to the Service status\n" + exception;
239         EmailSettings.getInstance().sendMessage(subject, message);
