67  /**
68   * Provides an application for handling features related to the administration
69   * system of the DOI server.
70   *
71   * The administration application provides the following features:
72   * <ul>
73   * <li>An asynchronous task to check the availability of created landing pages
74   * each {@value #PERIOD_SCHEDULER} days</li>
75   * <li>The form for creating DOI</li>
76   * <li>The Datacite status page to check Datacite services availability</li>
77   * <li>The Datacite stats page</li>
78   * <li>A service to create random string as a part of the DOI suffix</li>
79   * <li>A token service for creating token and to get token information</li>
80   * </ul>
81   * <p>
82   * <b>Security</b><br>
83   * --------------<br>
84   * The authentication is done by a simple login/password. Only users having the
85   * group "admin" will be allowed to log in to this application.<br>
86   * The website for creating DOI is opened on the net whereas the others services
87   * are filtered by IP. The allowed IPs are localhost and the IPs defined in
88   * {@value fr.cnes.doi.settings.Consts#ADMIN_IP_ALLOWER} attribute from the
89   * configuration file
90   *
91   * <p>
92   * <b>Routing</b><br>
93   * --------------<br>
94   * <br>
95   * root (DOI creation web form - no authorization)<br>
96   * |<br>
97   * |__ resources (status page - no authorization)<br>
98   * |__ status (Datacite status page - authorization)<br>
99   * |__ stats (Datacite stats page - authorization)<br>
100  * |_ ____________<br>
101  * |_|************|____ suffixProject (Get a random suffix - authorization)<br>
102  * |_|IP_filtering|____ token (Create a token - authorization)<br>
103  * |_|____________|____ token/{tokenID} (Get token information -
104  * authorization)<br>
105  *
106  * @see <a href="http://status.datacite.org">Datacite status page</a>
107  * @see <a href="https://stats.datacite.org/#tab-prefixes">Datacite stats
108  * page</a>
109  * @see SuffixProjectsResource Creating a project suffix for DOI
110  * @see TokenResource Creating a token and getting information about a token
111  * @author Jean-Christophe Malapert (jean-christophe.malapert@cnes.fr)
112  */

347     /**
348      * Creates a router for the AdminApplication. This router routes :
349      * <ul>
350      * <li>the web resources for the website with no authentication</li>
351      * <li>the REST resources for the system administration with
352      * authentication/authorization</li>
353      * </ul>
354      * The web resources are attached by default to the AdminApplication where
355      * as the REST resources for the system administration are attached with the
356      * {@link AdminApplication#ADMIN_URI}
357      *
358      * @see AdminApplication#createWebSiteRouter() the router that contains the
359      * the web resources
360      * @see AdminApplication#createAdminRouter() the router that contains the
361      * REST resources
362      * @see AdminApplication#createAuthenticator() the authentication mechanism
363      * @see AdminApplication#createRoleAuthorizer() the authorization mechanism
364      * @return Router
365      */
366     @Override
367     public Restlet createInboundRoot() {
368         LOG.traceEntry();
369         
370         final Filter logContext = new Filter() {
371             @Override
372             protected int beforeHandle(Request request, Response response) {
373                 final ClientInfo clientInfo = request.getClientInfo();
374                 final String ipAddress = request.getHeaders().getFirstValue(
375                         Consts.PROXIFIED_IP, clientInfo.getUpstreamAddress()
376                 );
377                 ThreadContext.put(Consts.LOG_IP_ADDRESS, ipAddress);
378                 return Filter.CONTINUE;
379             }
380         }; 
381         
382         // Defines the strategy of authentication (authentication is not required)
383         //   - authentication with login/pwd
384         final ChallengeAuthenticator challAuth = createAuthenticatorLoginBased();
385         challAuth.setOptional(true);
386 
387         //   - authentication with token
388         final ChallengeAuthenticator challTokenAuth = createTokenAuthenticator();
389         challTokenAuth.setOptional(false);
390         //if already authenticate 
391         challTokenAuth.setMultiAuthenticating(false);
392 
393         // Defines the authorization
394         final RoleAuthorizer authorizer = createRoleAuthorizer();
395 
396         // Defines the admin router as private
397         final String ips = DoiSettings.getInstance().getString(Consts.ADMIN_IP_ALLOWER);
398         final boolean isEnabled = ips != null;
399         final AllowerIP blocker = new AllowerIP(getContext(), isEnabled);
400 
401         // Defines the routers
402         final Router webSiteRouter = createWebSiteRouter();
403         final Router adminRouter = createAdminRouter();
404         
405         // pipeline of authentication and authorization
406         challAuth.setNext(challTokenAuth);
407         challTokenAuth.setNext(authorizer);
408         authorizer.setNext(blocker);
409         blocker.setNext(adminRouter);
410 
411         final Router router = new Router(getContext());
412         router.attachDefault(webSiteRouter);
413 
414         router.attach(ADMIN_URI, challAuth);
415         
416         logContext.setNext(router);        
417 
418         return LOG.traceExit(logContext);
419     }

421     /**
422      * Creates a authorization based on the role. Only users attached to the
423      * role {@link fr.cnes.doi.security.RoleAuthorizer#ROLE_ADMIN} are allowed
424      *
425      * @return the authorization that contains the access rights to the
426      * resources.
427      */
428     private RoleAuthorizer createRoleAuthorizer() {
429         LOG.traceEntry();
430 
431         final RoleAuthorizer roleAuth = new RoleAuthorizer() {
432 
433             /**
434              * Cancel verification on pre-flight OPTIONS method
435              *
436              * @param request request
437              * @param response response
438              * @return the result
439              */
440             @Override
441             public int beforeHandle(final Request request, final Response response) {
442                 final Method requestMethod = request.getMethod();
443                 final Reference requestReference = request.getOriginalRef();
444                 final String lastSeg = requestReference.getLastSegment();
445                 final String path = requestReference.getPath();
446                 boolean ignoreVerification = false;
447 
448                 if (requestMethod.equals(Method.OPTIONS)) {
449                     // options method is allowed independently of the role
450                     // just need to be authenticated
451                     ignoreVerification = true;
452                 } else if (!requestMethod.equals(Method.DELETE) && !requestMethod.equals(Method.GET)
453                         && requestReference.toString().contains(ADMIN_URI + TOKEN_URI)) {
454                     // token resource is allowed independently of the role
455                     // just need to be authenticated because we want to access to the 
456                     //authentication page of the GUI without authentication
457                     ignoreVerification = true;
458                 } else if (requestMethod.equals(Method.GET)) {
459                     // no check of the role when getting the list of projects, superusers and dois 
460                     if ("projects".equals(lastSeg) && requestReference.hasQuery()) {
461                         ignoreVerification = true;
462                     }
463                     // ignore method GET dois from project
464                     if ("dois".equals(lastSeg)) {
465                         ignoreVerification = true;
466                     }
467                     // Checks authentication for "users" group
468                     if (path.contains(ADMIN_URI + ROLE_USERS_URI)) {
469                         ignoreVerification = true;
470                     }
471                 }
472 
473                 final int status;
474                 if (ignoreVerification) {
475                     response.setStatus(Status.SUCCESS_OK);
476                     status = CONTINUE;
477                 } else {
478                     status = super.beforeHandle(request, response);
479                 }
480                 return status;
481             }
482 
483         };
484         roleAuth.setAuthorizedRoles(
485                 Arrays.asList(
486                         Role.get(this, fr.cnes.doi.security.RoleAuthorizer.ROLE_ADMIN)
487                 )
488         );
489 
490         return LOG.traceExit(roleAuth);
491     }

494      * Creates a router for REST services for the system administration. This
495      * router contains the following resources :
496      * <ul>
497      * <li>{@link AdminApplication#SUFFIX_PROJECT_URI} resource to handle the

499      * <li>{@link AdminApplication#TOKEN_URI} resource to handle the token for

502      *
503      * @see SuffixProjectsResource Resource to handle the the project suffix
504      * @see TokenResource Resource to handle the token resource
505      * @return the router
506      */

508         LOG.traceEntry();
509 
510         final Router router = new Router(getContext());
511 
512         router.attach(SUFFIX_PROJECT_URI, SuffixProjectsResource.class);
513 
514         router.attach(SUFFIX_PROJECT_URI + SUFFIX_PROJECT_NAME, ManageProjectsResource.class);
515         router.attach(SUFFIX_PROJECT_URI + SUFFIX_PROJECT_NAME + DOIS_URI,
516                 SuffixProjectsDoisResource.class);
517 
518         router.attach(SUFFIX_PROJECT_URI + SUFFIX_PROJECT_NAME + USERS_URI,
519                 ManageUsersResource.class);
520         router.attach(SUFFIX_PROJECT_URI + SUFFIX_PROJECT_NAME + USERS_URI + USERS_NAME,
521                 ManageUsersResource.class);
522 
523         router.attach(SUPERUSERS_URI, ManageSuperUsersResource.class);
524         router.attach(SUPERUSERS_URI + USERS_NAME, ManageSuperUserResource.class);
525 
526         router.attach(TOKEN_URI, TokenResource.class);
527         router.attach(TOKEN_URI + TOKEN_NAME_URI, TokenResource.class);
528 
529         router.attach(ROLE_USERS_URI, AuthenticateResource.class);
530         router.attach(ROLE_ADMIN_URI, AuthenticateResource.class);
531 
532         return LOG.traceExit(router);
533     }

581     /**
582      * Adds route attacURI to the target according to redirection mode.
583      *
584      * @param router router
585      * @param redirectorMode redirection mode
586      * @param target target
587      * @param attachURI attachURI
588      */
589     private void addServices(final Router router, final int redirectorMode, final String target,
590             final String attachURI) {
591         LOG.traceEntry("Parameters\n   router: {}\n   redirectorMode: {}\n   "
592                 + "target: {}\n  attachURI: {}",
593                 new JsonMessage(router), redirectorMode, target, attachURI);
594 
595         final Redirector redirector = new Redirector(getContext(), target, redirectorMode);
596 
597         final Filter authentication = new Filter() {
598             /**
599              * {@inheritDoc }
600              */
601             @Override
602             protected int doHandle(final Request request, final Response response) {
603                 response.setLocationRef(target);
604                 response.setStatus(Status.REDIRECTION_PERMANENT);
605                 return super.doHandle(request, response);
606             }
607         };
608         authentication.setNext(redirector);
609         router.attach(attachURI, authentication);
610         LOG.traceExit();
611     }
